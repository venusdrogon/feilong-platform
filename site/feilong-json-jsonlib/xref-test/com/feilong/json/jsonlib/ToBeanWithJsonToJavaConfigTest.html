126     /**
127      * Test to bean.
128      */
129     @Test
130     public void testToBean(){
131         String json = "{'data':[{'name':'get'},{'name':'set'}],'id':5}";
132         Map<String, Class<?>> classMap = newHashMap();
133         classMap.put("data", Person.class);
134 
135         MyBean myBean = JsonUtil.toBean(json, new JsonToJavaConfig(MyBean.class, classMap));
136 
137         assertThat(myBean, allOf(hasProperty("id", is(5L))));
138 
139         List<Object> data = myBean.getData();
140         assertThat(data.get(0), allOf(hasProperty("name", is("get"))));
141         assertThat(data.get(1), allOf(hasProperty("name", is("set"))));
142 
143     }
