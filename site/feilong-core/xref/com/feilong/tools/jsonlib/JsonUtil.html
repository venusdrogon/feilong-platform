271     /**
272      * 有些map 值很复杂,比如带有request信息, 这样的map转成json很可能由于一些级联关系而抛异常.
273      * 
274      * <h3>注意:</h3>
275      * 
276      * <blockquote>
277      * <ul>
278      * <li>此方法 将inputMap转成 simpleMap(<span style="color:red">原始inputMap不会变更</span>)</li>
279      * <li>此方法转换的simpleMap是 {@link TreeMap}类型,转换的json key经过排序的</li>
280      * </ul>
281      * </blockquote>
282      * 
283      * <h3>转换规则:</h3>
284      * 
285      * <blockquote>
286      * <ul>
287      * <li>如果value是isPrimitiveOrWrapper类型,那么会直接取到值设置到新的simpleMap中</li>
288      * <li>否则使用{@link String#valueOf(Object)}转换到simpleMap中</li>
289      * </ul>
290      * </blockquote>.
291      *
292      * @param <K>
293      *            the key type
294      * @param <V>
295      *            the value type
296      * @param inputMap
297      *            the input map
298      * @param allowFormatClassTypes
299      *            除了基本类型,数组之外允许的类型,请确保该类型可以被json format输出
300      * @return 如果 <code>inputMap</code> 是null,返回 {@link StringUtils#EMPTY}<br>
301      * @since 1.3.0
302      */
303     public static <K, V> String formatSimpleMap(Map<K, V> inputMap,Class<?>...allowFormatClassTypes){
304         if (null == inputMap){
305             return EMPTY;
306         }
307         Map<K, Object> simpleMap = new TreeMap<>();
308         for (Map.Entry<K, V> entry : inputMap.entrySet()){
309             V value = entry.getValue();
310             simpleMap.put(entry.getKey(), isAllowFormatType(value, allowFormatClassTypes) ? value : String.valueOf(value)); //注意 String.valueOf(value)如果value是null 那么会输出 "null"字符串
311         }
312         return format(simpleMap);
313     }

620     /**
621      * Format.
622      *
623      * @param obj
624      *            the obj
625      * @param jsonFormatConfig
626      *            the json format config
627      * @param indentFactor
628      *            the indent factor
629      * @param indent
630      *            the indent
631      * @return 如果 <code>obj</code> 是null,返回 {@link StringUtils#EMPTY}<br>
632      * @since 1.2.2
633      */
634     public static String format(Object obj,JsonFormatConfig jsonFormatConfig,int indentFactor,int indent){
635         if (null == jsonFormatConfig){
636             return format(obj, (JsonConfig) null, indentFactor, indent);
637         }
638         JsonConfig jsonConfig = getDefaultJsonConfig();
639 
640         //value处理器
641         Map<String, JsonValueProcessor> propertyNameAndJsonValueProcessorMap = jsonFormatConfig.getPropertyNameAndJsonValueProcessorMap();
642         if (isNotNullOrEmpty(propertyNameAndJsonValueProcessorMap)){
643             for (Map.Entry<String, JsonValueProcessor> entry : propertyNameAndJsonValueProcessorMap.entrySet()){
644                 jsonConfig.registerJsonValueProcessor(entry.getKey(), entry.getValue());
645             }
646         }
647         //排除
648         if (isNotNullOrEmpty(jsonFormatConfig.getExcludes())){
649             jsonConfig.setExcludes(jsonFormatConfig.getExcludes());
650         }
651         //包含
652         if (isNotNullOrEmpty(jsonFormatConfig.getIncludes())){
653             jsonConfig.setJsonPropertyFilter(new ArrayContainsPropertyNamesPropertyFilter(jsonFormatConfig.getIncludes()));
654         }
655 
656         return format(obj, jsonConfig, indentFactor, indent);
657     }

759      * @see net.sf.json.JsonConfig#registerJsonValueProcessor(Class, JsonValueProcessor)
760      * @see org.apache.commons.collections4.IteratorUtils#toList(Iterator)
761      * @see org.apache.commons.collections4.IteratorUtils#toList(Iterator, int)
762      * @see net.sf.json.JSONSerializer#toJSON(Object)

764     static JSON toJSON(Object obj,JsonConfig jsonConfig){
765         JsonConfig useJsonConfig = defaultIfNull(jsonConfig, DEFAULT_JSON_CONFIG);

794     }

1230      * <b>返回:</b>
1231      * 
1232      * <pre class="code">
1233        {"mybean":{
1234                  "id": 0,
1235                  "data": [{
1236                      "name": "get"
1237                  }]
1238        }}
1239      * </pre>
1240      * 
1241      * </blockquote>
1242      *
1243      * @param <T>
1244      *            the generic type
1245      * @param json
1246      *            e.g. {'mybean':{'data':[{'name':'get'}]}}
1247      * @param rootClass
1248      *            e.g. MyBean.class ,see {@link net.sf.json.JsonConfig#setRootClass(Class)}
1249      * @param classMap
1250      *            e.g. classMap.put("data", Person.class)
1251      * @return 如果 <code>json</code> 是null或者empty,返回 {@link Collections#emptyMap()}<br>
1252      *         如果 <code>rootClass</code> 是null,那么直接将json里面的value 作为map 的value
1253      * @see net.sf.json.JSONObject#keys()
1254      * @see #toBean(Object, Class, Map)
1255      * @since 1.9.2 use LinkedHashMap instead of HashMap
1256      */
1257     @SuppressWarnings("unchecked")
1258     public static <T> Map<String, T> toMap(String json,Class<T> rootClass,Map<String, Class<?>> classMap){
1259         LOGGER.trace("input json:[{}],rootClass:[{}]", json, rootClass);
1260 
1261         if (isNullOrEmpty(json)){
1262             return emptyMap();
1263         }
1264 
1265         Map<String, T> map = new LinkedHashMap<>();
1266 
1267         JSONObject jsonObject = toJSONObject(json);
1268         Iterator<String> keys = jsonObject.keys();
1269         while (keys.hasNext()){
1270             String key = keys.next();
1271             Object value = jsonObject.get(key);
1272             LOGGER.trace("key:[{}],value:[{}],value type is:[{}]", key, value, value.getClass().getName());
1273             map.put(key, null == rootClass ? (T) value : toBean(value, rootClass, classMap));//如果rootClass是null,表示不需要转换
1274         }
1275         return map;
1276     }

1323     /**
1324      * 从json串转换成实体对象,并且实体集合属性存有另外实体Bean.
1325      * 
1326      * <h3>示例:</h3>
1327      * <blockquote>
1328      * 
1329      * <pre class="code">
1330      * String json = "{'data':[{'name':'get'},{'name':'set'}],'id':5}";
1331      * Map{@code <String, Class<?>>} classMap = new HashMap{@code <String, Class<?>>}();
1332      * classMap.put("data", Person.class);
1333      * 
1334      * MyBean myBean = JsonUtil.toBean(json, MyBean.class, classMap);
1335      * LOGGER.info(JsonUtil.format(myBean));
1336      * </pre>
1337      * 
1338      * <b>返回:</b>
1339      * 
1340      * <pre class="code">
1341      {
1342              "id": 5,
1343              "data":[{
1344                      "dateAttr": null,
1345                      "name": "get"
1346                  },{
1347                      "dateAttr": null,
1348                      "name": "set"
1349                  }
1350              ]
1351       }
1352      * </pre>
1353      * 
1354      * </blockquote>
1355      *
1356      * @param <T>
1357      *            the generic type
1358      * @param json
1359      *            e.g. {'data':[{'name':'get'},{'name':'set'}]}
1360      * @param rootClass
1361      *            e.g. MyBean.class,see {@link net.sf.json.JsonConfig#setRootClass(Class)}
1362      * @param classMap
1363      *            e.g. classMap.put("data", Person.class)
1364      * @return 如果<code>json</code> 是null,那么返回 null
1365      * @see #toBean(Object, JsonConfig)
1366      * @see JSONObject#fromObject(Object, JsonConfig)
1367      * @see net.sf.json.JsonConfig#setRootClass(Class)
1368      */
1369     public static <T> T toBean(Object json,Class<T> rootClass,Map<String, Class<?>> classMap){
1370         if (null == json){
1371             return null;
1372         }
1373         JSONObject jsonObject = JSONObject.fromObject(json);
1374 
1375         JsonConfig jsonConfig = getDefaultJsonConfig();
1376         jsonConfig.setRootClass(rootClass);
1377 
1378         if (isNotNullOrEmpty(classMap)){
1379             jsonConfig.setClassMap(classMap);
1380         }
1381         return toBean(jsonObject, jsonConfig);
1382     }

1387      * @param <T>

1389      * @param json

1391      * @param jsonConfig
1392      *            the json config
1393      * @return 如果<code>json</code> 是null,那么返回 null
1394      * @see net.sf.json.JSONObject#toBean(JSONObject, JsonConfig)

1396     @SuppressWarnings("unchecked")
1397     private static <T> T toBean(Object json,JsonConfig jsonConfig){
1398         JSONObject jsonObject = JSONObject.fromObject(json);

1446     /**
1447      * 是否是可以被json format的类型.
1448      *
1449      * @param <V>
1450      *            the value type
1451      * @param value
1452      *            the value

1454      *            the allow class types

1460             return true;

1466                         || ClassUtil.isInstanceAnyClass(value, allowClassTypes)//
