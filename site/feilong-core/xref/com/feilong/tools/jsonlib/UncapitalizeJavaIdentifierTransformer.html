16  package com.feilong.tools.jsonlib;
17  
18  import org.apache.commons.lang3.StringUtils;
19  
20  import net.sf.json.util.JavaIdentifierTransformer;
21  
22  /**
23   * 当json转成bean的时候,json字符串里面的属性名字可能首字母是大写的,转成bean里面属性首字母小写的转换器.
24   * 
25   * <p>
26   * 比如 MemberId,但是我们的java bean里面的属性名字是标准的 驼峰命名法则,比如 memberId
27   * </p>
28   * 
29   * <h3>示例:</h3>
30   * 
31   * <blockquote>
32   * 
33   * <p>
34   * <b>场景:</b> 从相关接口得到的json数据格式如下(注意:属性首字母是大写的):
35   * </p>
36   * 
37   * <pre class="code">
38   * {
39   * "MemberNo":"11105000009",
40   * "Name":null,
41   * "Gender":"",
42   * "Phone":"15036334567",
43   * "Email":null,
44   * "Birthday":""
45   * }
46   * </pre>
47   * 
48   * <p>
49   * 但是我们的类是标准的java bean,属性符合驼峰命名规则,比如:
50   * </p>
51   * 
52   * <pre class="code">
53   * public class CrmMemberInfoCommand{
54   * 
55   *     //** 会员编号
56   *     private String memberNo;
57   * 
58   *     //** 姓名/昵称 
59   *     private String name;
60   * 
61   *     //** 电话号码 
62   *     private String phone;
63   * 
64   *     //** 性别：男；女 
65   *     private String gender;
66   * 
67   *     //** 生日：Format：yyyy-MM-dd 
68   *     private String birthday;
69   * 
70   *     //** 邮箱 
71   *     private String email;
72   * 
73   *     // setter getter
74   * }
75   * </pre>
76   * 
77   * <p>
78   * 此时可以使用该类,示例如下:
79   * </p>
80   * 
81   * <pre class="code">
82   * 
83   * public void testToBean2(){
84   *     String json = "{'MemberNo':'11105000009','Name':null,'Gender':'','Phone':'15036334567','Email':null,'Birthday':''}";
85   * 
86   *     JsonToJavaConfig jsonToJavaConfig = new JsonToJavaConfig();
87   *     jsonToJavaConfig.setRootClass(CrmMemberInfoCommand.class);
88   *     jsonToJavaConfig.setJavaIdentifierTransformer(UncapitalizeJavaIdentifierTransformer.UNCAPITALIZE);
89   * 
90   *     CrmMemberInfoCommand crmMemberInfoCommand = JsonUtil.toBean(json, jsonToJavaConfig);
91   *     //.....
92   * }
93   * 
94   * </pre>
95   * 
96   * </blockquote>
97   *
98   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
99   * @see <a href="https://github.com/venusdrogon/feilong-core/issues/509">issue 509</a>
100  * @since 1.9.4
101  */
102 public class UncapitalizeJavaIdentifierTransformer extends JavaIdentifierTransformer{
103 
104     /** 首字母小写 transformer 'MemberNo' {@code =>} 'memberNo'. */
105     public static final JavaIdentifierTransformer UNCAPITALIZE = new UncapitalizeJavaIdentifierTransformer();
106 
107     /*
108      * (non-Javadoc)
109      * 
110      * @see net.sf.json.util.JavaIdentifierTransformer#transformToJavaIdentifier(java.lang.String)
111      */
112     @Override
113     public String transformToJavaIdentifier(String str){
114         return StringUtils.uncapitalize(str);
115     }
116 }
