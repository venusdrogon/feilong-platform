16  package com.feilong.core.util;
17  
18  import static com.feilong.core.Validator.isNullOrEmpty;
19  
20  import java.util.Enumeration;
21  
22  import org.apache.commons.collections4.IteratorUtils;
23  import org.apache.commons.collections4.iterators.EnumerationIterator;
24  
25  /**
26   * {@link Enumeration}工具类.
27   * 
28   * <p>
29   * {@link Enumeration}接口是JDK 1.0时就推出的,是最早的迭代输出接口,最早使用Vector时就是使用{@link Enumeration}接口进行输出的。<br>
30   * 虽然{@link Enumeration}是一个旧的类,但是在JDK1.5之后为Enumeration类进行了扩充,增加了泛型的操作应用
31   * </p>
32   * 
33   * <h3>为什么还要继续使用{@link Enumeration}?</h3>
34   * 
35   * <blockquote>
36   * <p>
37   * {@link Enumeration}和Iterator接口的功能非常类似,而且{@link Enumeration}接口中方法的名称也比 接口中的方法名称长很多,那为什么还要继续使用{@link Enumeration}呢？
38   * </p>
39   * <p>
40   * 在旧的操作中依然会使用{@link Enumeration}接口。<br>
41   * 实际上Java的发展经历了很长的时间,一些比较古老的系统或是类库的方法中(例如,本系列的下一步Web开发中就存在这样的操作方法)还在使用{@link Enumeration}接口,所以掌握其操作也是很有必要的。
42   * </p>
43   * <p>
44   * 而Iterator是JDK1.2才添加的接口,它也是为了HashMap、ArrayList等集合提供遍历接口。<br>
45   * Iterator是支持fail-fast机制的:当多个线程对同一个集合的内容进行操作时,就可能会产生fail-fast事件。
46   * </p>
47   * </blockquote>
48   *
49   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
50   * @see org.apache.commons.collections4.EnumerationUtils
51   * @since 1.5.3
52   */
53  public final class EnumerationUtil{
54  
55      /** Don't let anyone instantiate this class. */
56      private EnumerationUtil(){
57          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
58          //see 《Effective Java》 2nd
59          throw new AssertionError("No " + getClass().getName() + " instances for you!");
60      }
61  
62      //---------------------------------------------------------------
63  
64      /**
65       * 判断<code>enumeration</code>枚举里面,是否有指定的元素<code>value</code>.
66       * 
67       * <h3>示例:</h3>
68       * 
69       * <blockquote>
70       * 
71       * <pre class="code">
72       * EnumerationUtil.contains(null, "a")                                          =   false
73       * 
74       * EnumerationUtil.contains(toEnumeration(toList("4", "5")), "a")               =   false
75       * EnumerationUtil.contains(toEnumeration(toList("4", "5")), "4")               =   true
76       * EnumerationUtil.contains(toEnumeration(toList("4", "5", "")), "")            =   true
77       * EnumerationUtil.contains(toEnumeration(toList("4", "5", "", null)), null)    =   true
78       * </pre>
79       * 
80       * </blockquote>
81       *
82       * @param <O>
83       *            the generic type
84       * @param enumeration
85       *            the enumeration
86       * @param value
87       *            指定的元素
88       * @return 如果 <code>enumeration</code> 是null或者empty,返回 false<br>
89       *         否则如果 contains 返回true,<br>
90       *         其他返回false
91       * @see "org.springframework.util.CollectionUtils#contains(Enumeration, Object)"
92       * @see org.apache.commons.collections4.iterators.EnumerationIterator
93       * @see org.apache.commons.collections4.IteratorUtils#contains(java.util.Iterator, Object)
94       */
95      public static <O> boolean contains(Enumeration<O> enumeration,O value){
96          return isNullOrEmpty(enumeration) ? false : IteratorUtils.contains(new EnumerationIterator<O>(enumeration), value);
97      }
98  }
