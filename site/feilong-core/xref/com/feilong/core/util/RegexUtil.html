56      /**
57       * 编译给定正则表达式 <code>regexPattern</code> ,并尝试将给定输入 <code>input</code> 与其匹配.
58       * 
59       * <p>
60       * {@link Pattern#matches(String, CharSequence)} 等价于{@link #getMatcher(String, CharSequence)}.matches();
61       * </p>
62       *
63       * @param regexPattern
64       *            正则表达式字符串,pls use {@link RegexPattern}
65       * @param input
66       *            The character sequence to be matched,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
67       * @return 如果input 符合 regex的正则表达式格式,返回true, 否则返回 false;<br>
68       *         如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
69       *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
70       * @see #getMatcher(String, CharSequence)
71       * @see Matcher#matches()
72       * @see Pattern#matches(String, CharSequence)
73       * @since 1.0.7
74       */
75      public static boolean matches(String regexPattern,CharSequence input){
76          return getMatcher(regexPattern, input).matches();
77      }

81      /**
82       * 返回在以前匹配操作期间由给定组捕获的输入子序列.
83       * 
84       * <p>
85       * 对于匹配器 m、输入序列 s 和组索引 g,表达式 m.group(g) 和 s.substring(m.start(g), m.end(g))是等效的.<br>
86       * 捕获组是从 1开始从左到右的索引.组0表示整个模式,因此表达式 m.group(0)等效于 m.group().
87       * </p>
88       * 
89       * <h3>示例:</h3>
90       * 
91       * <blockquote>
92       * 
93       * <pre class="code">
94       * String regexPattern = "(.*?)@(.*?)";
95       * String email = "feilong@163.com";
96       * 
97       * RegexUtil.group(regexPattern, email);
98       * </pre>
99       * 
100      * <b>返回:</b>
101      * 
102      * <pre class="code">
103      *    0 feilong@163.com
104      *    1 feilong
105      *    2 163.com
106      * </pre>
107      * 
108      * </blockquote>
109      * 
110      * @param regexPattern
111      *            正则表达式模式,比如 (.*?)@(.*?)
112      * @param input
113      *            需要被group的字符串,比如 feilong@163.com,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
114      * @return 如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
115      *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
116      *         如果 匹配不了,返回 {@link java.util.Collections#emptyMap()}
117      * @see #getMatcher(String, CharSequence)
118      * @see Matcher#group(int)
119      * @since 1.0.7
120      */
121     public static Map<Integer, String> group(String regexPattern,CharSequence input){
122         Matcher matcher = getMatcher(regexPattern, input);
123         if (!matcher.matches()){
124             LOGGER.trace("[not matches] ,\n\tregexPattern:[{}] \n\tinput:[{}]", regexPattern, input);
125             return emptyMap();
126         }
127 
128         //---------------------------------------------------------------
129         int groupCount = matcher.groupCount();
130         Map<Integer, String> map = newLinkedHashMap(groupCount + 1);
131         for (int i = 0; i <= groupCount; ++i){
132             //匹配的索引
133             String groupValue = matcher.group(i); //map.put(0, matcher.group());// 捕获组是从 1 开始从左到右的索引.组0表示整个模式,因此表达式 m.group(0) 等效于 m.group().
134             LOGGER.trace("matcher group[{}],start-end:[{}-{}],groupValue:[{}]", i, matcher.start(i), matcher.end(i), groupValue);
135             map.put(i, groupValue);//groupValue
136         }
137 
138         //---------------------------------------------------------------
139 
140         if (LOGGER.isTraceEnabled()){
141             LOGGER.trace("regexPattern:[{}],input:[{}],groupMap:{}", regexPattern, input, map);
142         }
143         return map;
144     }

146     /**
147      * 返回在以前匹配操作期间由给定组捕获的输入子序列.
148      * 
149      * <p>
150      * 对于匹配器 m、输入序列 s 和组索引 g,表达式 m.group(g) 和 s.substring(m.start(g), m.end(g))是等效的.<br>
151      * 捕获组是从 1开始从左到右的索引.组0表示整个模式,因此表达式 m.group(0)等效于 m.group().
152      * </p>
153      * 
154      * <h3>示例:</h3>
155      * 
156      * <blockquote>
157      * 
158      * <pre class="code">
159      * 
160      * String regexPattern = "(.*?)@(.*?)";
161      * String email = "feilong@163.com";
162      * 
163      * RegexUtil.group(regexPattern, email, 1);<span style="color:green">//feilong</span>
164      * RegexUtil.group(regexPattern, email, 2);<span style="color:green">//163.com</span>
165      * 
166      * </pre>
167      * 
168      * </blockquote>
169      *
170      * @param regexPattern
171      *            正则表达式模式,比如 (.*?)@(.*?)
172      * @param input
173      *            需要被group的字符串,比如 feilong@163.com,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
174      * @param groupNo
175      *            组号,从0开始
176      * @return 如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
177      *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
178      *         如果 {@code input < 0} ,抛出 {@link IllegalArgumentException}<br>
179      * @see #getMatcher(String, CharSequence)
180      * @see Matcher#group(int)
181      * @since 1.0.7
182      */
183     public static String group(String regexPattern,CharSequence input,int groupNo){
184         Validate.isTrue(groupNo >= 0, "groupNo must >=0");
185 
186         Map<Integer, String> map = group(regexPattern, input);
187         return map.get(groupNo);
188     }

191     /**
192      * Gets the matcher.
193      *
194      * @param regexPattern
195      *            正则表达式字符串,pls use {@link RegexPattern}
196      * @param input
197      *            The character sequence to be matched,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on

200      * @see Pattern#compile(String)
201      * @since 1.0.7

204         return getMatcher(regexPattern, input, 0);
205     }

207     /**
208      * Gets the matcher.
209      *
210      * @param regexPattern
211      *            正则表达式字符串,pls use {@link RegexPattern}
212      * @param input
213      *            The character sequence to be matched,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
214      * @param flags
215      *            如果需要多个组合,可以使用罗辑或
216      *            <blockquote><code>Pattern.compile(regex, CASE_INSENSITIVE | DOTALL);</code></blockquote>
217      *            <ul>
218      *            <li>{@link Pattern#CASE_INSENSITIVE} 匹配字符时与大小写无关,该标志默认只考虑US ASCII字符.</li>
219      *            <li>{@link Pattern#MULTILINE} ^和$匹配一行的开始和结尾,而不是整个输入</li>
220      *            <li>{@link Pattern#UNICODE_CASE} 当与CASE_INSENSITIVE结合时,使用Unicode字母匹配</li>
221      *            <li>{@link Pattern#CANON_EQ} 考虑Unicode字符的规范等价</li>
222      *            <li>{@link Pattern#DOTALL} 当使用此标志时,.符号匹配包括行终止符在内的所有字符</li>
223      *            <li>{@link Pattern#UNIX_LINES} 当在多行模式下匹配^和$时,只将'\n'看作行终止符</li>
224      *            <li>{@link Pattern#LITERAL} 启用模式的字面值解析.</li>
225      *            <li>{@link Pattern#COMMENTS} 模式中允许空白和注释. <br>
226      *            此模式将忽略空白和在结束行之前以 # 开头的嵌入式注释. <br>
227      *            通过嵌入式标志表达式 (?x) 也可以启用注释模式. <br>
228      *            </li>
229      *            </ul>
230      * @return 如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
231      *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
232      * @see Pattern#compile(String, int)
233      * @since 1.5.3
234      */
235     private static Matcher getMatcher(String regexPattern,CharSequence input,int flags){
236         Validate.notNull(regexPattern, "regexPattern can't be null!");
237         Validate.notNull(input, "input can't be null!");
238 
239         Pattern pattern = Pattern.compile(regexPattern, flags);
240         return pattern.matcher(input);
241     }
