16  package com.feilong.core.util.equator;
17  
18  import java.io.Serializable;
19  
20  import org.apache.commons.collections4.Equator;
21  import org.apache.commons.lang3.StringUtils;
22  
23  /**
24   * 忽视大小写的实现.
25   * 
26   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
27   * @see org.apache.commons.collections4.functors.DefaultEquator
28   * @since 1.10.1
29   */
30  public class IgnoreCaseEquator implements Equator<String>,Serializable{
31  
32      /** The Constant serialVersionUID. */
33      private static final long             serialVersionUID = 5700113137952086493L;
34  
35      /** Static instance. */
36      // the static instance works for all types
37      public static final IgnoreCaseEquator INSTANCE         = new IgnoreCaseEquator();
38  
39      //---------------------------------------------------------------
40  
41      /**
42       * Restricted constructor.
43       */
44      private IgnoreCaseEquator(){
45          super();
46      }
47  
48      //---------------------------------------------------------------
49  
50      /*
51       * (non-Javadoc)
52       * 
53       * @see org.apache.commons.collections4.Equator#equate(java.lang.Object, java.lang.Object)
54       */
55      @Override
56      public boolean equate(final String s1,final String s2){
57          return StringUtils.equalsIgnoreCase(s1, s2);
58      }
59  
60      //---------------------------------------------------------------
61  
62      /**
63       * {@inheritDoc}
64       *
65       * @return <code>s.hashCode()</code> if <code>s</code> is non-<code>null</code>, else -1.
66       */
67      @Override
68      public int hash(final String s){
69          return s == null ? -1 : s.hashCode();
70      }
71  
72  }
