50      /**
51       * 按照不同指定属性 <code>propertyNameAndOrders</code> 排序的 {@link Comparator}.
52       *
53       * @param <T>
54       *            the generic type
55       * @param propertyNameAndOrders
56       *            属性名称和排序因子,
57       * 
58       *            <p>
59       *            格式可以是纯的属性名称, 比如 "name"; 也可以是属性名称+排序因子(以空格分隔),比如 "name desc"
60       *            </p>
61       * 
62       *            <h3>说明:</h3>
63       *            <blockquote>
64       * 
65       *            <dl>
66       *            <dt>关于属性名称</dt>
67       *            <dd>
68       *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be
69       *            modified,参见<a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,<br>
70       *            该属性对应的value 必须实现 {@link Comparable}接口.
71       *            </dd>
72       * 
73       *            <dt>关于排序因子</dt>
74       *            <dd>
75       *            可以没有排序因子<br>
76       *            如果有,值可以是asc(顺序),desc(倒序)两种;<br>
77       *            如果没有,默认按照asc(顺序)排序;<br>
78       *            此外,asc/desc忽略大小写
79       *            </dd>
80       * 
81       *            </dl>
82       * 
83       *            </blockquote>
84       * @return 如果propertyNameAndOrders是单值,那么直接调用 {@link #propertyComparator(String)} 返回
85       * @throws NullPointerException
86       *             如果 <code>propertyNameAndOrders</code> 是null,<br>
87       *             或者有元素是 null;
88       * @throws IllegalArgumentException
89       *             如果 <code>propertyNameAndOrders</code> 是empty,<br>
90       *             或者有元素是 blank
91       * @see org.apache.commons.collections4.ComparatorUtils#chainedComparator(java.util.Collection)
92       * @since 1.10.2 support propertyNameAndOrder
93       */
94      public static <T> Comparator<T> chainedComparator(String...propertyNameAndOrders){
95          Validate.notEmpty(propertyNameAndOrders, "propertyNameAndOrders can't be null/empty!");
96  
97          //如果propertyNameAndOrders是单值,那么直接调用 com.feilong.core.util.comparator.BeanComparatorUtil.propertyComparator(String) 返回
98          if (1 == propertyNameAndOrders.length){
99              return propertyComparator(propertyNameAndOrders[0]);
100         }
101 
102         //---------------------------------------------------------------
103 
104         List<Comparator<T>> comparators = newArrayList();
105         for (String propertyNameAndOrder : propertyNameAndOrders){
106             Validate.notBlank(propertyNameAndOrder, "propertyNameAndOrder can't be blank!");
107 
108             String[] propertyNameAndOrderArray = SortHelper.parsePropertyNameAndOrder(propertyNameAndOrder);
109 
110             //注意:此处不要使用 propertyComparator(propertyName)
111 
112             //因为,PropertyComparator 如果属性值相同,会使用其他规则继续比较(为了TreeMap/treeSet), 
113             //也就是说,通常而言一次就比较出顺序,后续的propertyNameAndOrders 就没作用了
114 
115             Comparator instance = ComparatorUtils.nullHighComparator(ComparableComparator.comparableComparator()); //null排在最后面  
116 
117             BeanComparator<T> beanComparator = new BeanComparator<>(propertyNameAndOrderArray[0], instance);
118             comparators.add(isAsc(propertyNameAndOrderArray) ? beanComparator : reversedComparator(beanComparator));
119         }
120         return ComparatorUtils.chainedComparator(comparators);
121     }

125     /**
126      * 指定属性 <code>propertyName</code> ,按照自然顺序 排序的 {@link Comparator}.
127      *
128      * @param <T>
129      *            the generic type
130      * @param propertyNameAndOrder
131      *            属性名称和排序因子,
132      * 
133      *            <p>
134      *            格式可以是纯的属性名称, 比如 "name"; 也可以是属性名称+排序因子(以空格分隔),比如 "name desc"
135      *            </p>
136      * 
137      *            <h3>说明:</h3>
138      *            <blockquote>
139      * 
140      *            <dl>
141      *            <dt>关于属性名称</dt>
142      *            <dd>
143      *            泛型T对象的属性名称,Possibly indexed and/or nested name of the property to be
144      *            modified,参见<a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,<br>
145      *            该属性对应的value 必须实现 {@link Comparable}接口.
146      *            </dd>
147      * 
148      *            <dt>关于排序因子</dt>
149      *            <dd>
150      *            可以没有排序因子<br>
151      *            如果有,值可以是asc(顺序),desc(倒序)两种;<br>
152      *            如果没有,默认按照asc(顺序)排序;<br>
153      *            此外,asc/desc忽略大小写
154      *            </dd>
155      * 
156      *            </dl>
157      * 
158      *            </blockquote>
159      * @return {@link PropertyComparator}
160      * @throws NullPointerException
161      *             如果 <code>propertyNameAndOrder</code> 是null
162      * @throws IllegalArgumentException
163      *             如果 <code>propertyNameAndOrder</code> 是blank
164      * @see PropertyComparator#PropertyComparator(String)
165      */
166     public static <T> Comparator<T> propertyComparator(String propertyNameAndOrder){
167         Validate.notBlank(propertyNameAndOrder, "propertyNameAndOrder can't be blank!");
168 
169         String[] propertyNameAndOrderArray = SortHelper.parsePropertyNameAndOrder(propertyNameAndOrder);
170 
171         PropertyComparator<T> propertyComparator = new PropertyComparator<>(propertyNameAndOrderArray[0]);
172         return isAsc(propertyNameAndOrderArray) ? propertyComparator : reversedComparator(propertyComparator);
173     }

175     /**
176      * 指定属性 <code>propertyName</code> 按照固定顺序值 <code>propertyValues</code> 排序的 {@link Comparator}.
177      *
178      * @param <T>
179      *            the generic type
180      * @param <V>
181      *            the value type
182      * @param propertyName
183      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
184      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,该属性对应的value 必须实现 {@link Comparable}接口.
185      * @param propertyValues
186      *            the property values
187      * @return the comparator
188      * @throws NullPointerException
189      *             如果 <code>propertyName</code> 是null,或者<code>propertyValues</code> 是null
190      * @throws IllegalArgumentException
191      *             如果 <code>propertyName</code> 是blank
192      * @see #propertyComparator(String, List)
193      */
194     @SafeVarargs
195     public static <T, V> Comparator<T> propertyComparator(String propertyName,V...propertyValues){
196         Validate.notBlank(propertyName, "propertyName can't be blank!");
197         Validate.notNull(propertyValues, "propertyValues can't be null!");
198         return propertyComparator(propertyName, toList(propertyValues));
199     }

201     /**
202      * 指定属性 <code>propertyName</code> 按照固定顺序值 <code>propertyValues</code> 排序的 {@link Comparator}.
203      * 
204      * <p>
205      * 调用 {@link #propertyComparator(String, List, UnknownObjectBehavior)},默认 {@link UnknownObjectBehavior#AFTER}
206      * </p>
207      *
208      * @param <T>
209      *            the generic type
210      * @param <V>
211      *            the value type
212      * @param propertyName
213      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
214      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,该属性对应的value 必须实现 {@link Comparable}接口.
215      * @param propertyValues
216      *            the property values
217      * @return the comparator
218      * @throws NullPointerException
219      *             如果 <code>propertyName</code> 是null,或者<code>propertyValues</code> 是null
220      * @throws IllegalArgumentException
221      *             如果 <code>propertyName</code> 是blank
222      * @see #propertyComparator(String, List, UnknownObjectBehavior)
223      */
224     public static <T, V> Comparator<T> propertyComparator(String propertyName,List<V> propertyValues){
225         Validate.notBlank(propertyName, "propertyName can't be blank!");
226         Validate.notNull(propertyValues, "propertyValues can't be null!");
227         return propertyComparator(propertyName, propertyValues, UnknownObjectBehavior.AFTER);
228     }

230     /**
231      * 指定属性 <code>propertyName</code> 按照固定顺序值 <code>propertyValues</code>,并且可以指定 <code>unknownObjectBehavior</code> 排序的 {@link Comparator}.
232      *
233      * @param <T>
234      *            the generic type
235      * @param <V>
236      *            the value type
237      * @param propertyName
238      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
239      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,该属性对应的value 必须实现 {@link Comparable}接口.
240      * @param propertyValues
241      *            the property values
242      * @param unknownObjectBehavior
243      *            the unknown object behavior
244      * @return the comparator
245      * @throws NullPointerException
246      *             如果 <code>propertyName</code> 是null,或者<code>propertyValues</code> 是null
247      * @throws IllegalArgumentException
248      *             如果 <code>propertyName</code> 是blank
249      * @see PropertyComparator#PropertyComparator(String, Comparator)
250      * @see FixedOrderComparator#FixedOrderComparator(List)
251      * @since 1.8.2
252      */
253     public static <T, V> Comparator<T> propertyComparator(
254                     String propertyName,
255                     List<V> propertyValues,
256                     UnknownObjectBehavior unknownObjectBehavior){
257         Validate.notBlank(propertyName, "propertyName can't be blank!");
258         Validate.notNull(propertyValues, "propertyValues can't be null!");
259         FixedOrderComparator<V> fixedOrderComparator = new FixedOrderComparator<>(propertyValues);
260         fixedOrderComparator.setUnknownObjectBehavior(unknownObjectBehavior);
261         return new PropertyComparator<>(propertyName, fixedOrderComparator);
262     }
