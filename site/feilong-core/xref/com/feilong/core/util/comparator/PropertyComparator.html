30  /**
31   * 属性比较器,自动获取 <code>T</code>中的属性名字是 {@link #propertyName}的值,进行比较,不用每个需要排序的字段创建 {@link Comparator}类.
32   * 
33   * <h3>关于 {@link #propertyName}:</h3>
34   * 
35   * <blockquote>
36   * <p>
37   * {@link #propertyName} 取出来的值,必须实现 {@link Comparable}接口,比如 {@link Integer}, {@link String}等类型
38   * </p>
39   * </blockquote>
40   * 
41   * <h3>顺序:</h3>
42   * 
43   * <blockquote>
44   * <p>
45   * 该类默认是<span style="color:red">正序</span>的形式,如果需要反序,请再使用 {@link ReverseComparator}进行包装
46   * </p>
47   * </blockquote>
48   *
49   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
50   * @param <T>
51   *            the generic type
52   * @see org.apache.commons.collections4.ComparatorUtils
53   * @see "org.springframework.beans.support.PropertyComparator"
54   * @see org.apache.commons.beanutils.BeanComparator
55   * @see org.apache.commons.collections4.comparators.BooleanComparator
56   * @see org.apache.commons.collections4.comparators.ReverseComparator
57   * @see org.apache.commons.collections4.comparators.ComparableComparator
58   * @see <a href=
59   *      "http://stackoverflow.com/questions/19325256/java-lang-illegalargumentexception-comparison-method-violates-its-general-contr">java-
60   *      lang-illegalargumentexception-comparison-method-violates-its-general-contr</a>
61   * @see <a href=
62   *      "http://stackoverflow.com/questions/11441666/java-error-comparison-method-violates-its-general-contract">Java error: Comparison
63   *      method violates its general contract</a>
64   * @since 1.2.0
65   */

172     /**
173      * The Constructor.
174      * 
175      * <h3>使用场景:</h3>
176      * <blockquote>
177      * 诸如需要排序的对象指定属性类型是数字,但是申明类型的时候由于种种原因是字符串,<br>
178      * 此时需要排序,如果不转成Integer比较的话,字符串比较13 将会在 3数字的前面
179      * </blockquote>
180      * 
181      * <h3>示例:</h3>
182      * <blockquote>
183      * 
184      * <p>
185      * 我们现在有这样的数据,其中属性 totalNo是字符串类型的
186      * </p>
187      * 
188      * <pre class="code">
189      * CourseEntity courseEntity1 = new CourseEntity();
190      * courseEntity1.setTotalNo("3");
191      * 
192      * CourseEntity courseEntity2 = new CourseEntity();
193      * courseEntity2.setTotalNo("13");
194      * 
195      * List{@code <CourseEntity>} courseList = new ArrayList{@code <>}();
196      * courseList.add(courseEntity1);
197      * courseList.add(courseEntity2);
198      * </pre>
199      * 
200      * 如果 我们只是使用 propertyName进行排序的话:
201      * 
202      * <pre class="code">
203      * Collections.sort(courseList, new PropertyComparator{@code <CourseEntity>}("totalNo"));
204      * LOGGER.debug(JsonUtil.format(courseList));
205      * </pre>
206      * 
207      * 那么<b>返回:</b>
208      * 
209      * <pre class="code">
210      * [{
211      * "totalNo": "13",
212      * "name": ""
213      * },{
214      * "totalNo": "3",
215      * "name": ""
216      * }]
217      * </pre>
218      * 
219      * 如果我们使用 propertyName+ propertyValueConvertToClass进行排序的话:
220      * 
221      * <pre class="code">
222      * Collections.sort(courseList, new PropertyComparator{@code <CourseEntity>}("totalNo", Integer.class));
223      * LOGGER.debug(JsonUtil.format(courseList));
224      * </pre>
225      * 
226      * <b>返回:</b>
227      * 
228      * <pre class="code">
229      * [{
230      * "totalNo": "3",
231      * "name": ""
232      * },{
233      * "totalNo": "13",
234      * "name": ""
235      * }]
236      * </pre>
237      * 
238      * </blockquote>
239      *
240      * @param propertyName
241      *            指定bean对象排序属性名字(默认正序).
242      * 
243      *            <p>
244      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
245      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,<br>
246      *            该属性的value 必须实现 {@link Comparable}接口.
247      *            </p>
248      * @param propertyValueConvertToClass
249      *            反射提取出来的值,需要类型转成到的类型
250      * @since 1.5.0
251      */
252     @SuppressWarnings("rawtypes")
253     public PropertyComparator(String propertyName, Class<? extends Comparable> propertyValueConvertToClass){
254         Validate.notBlank(propertyName, "propertyName can't be blank!");
255         this.propertyName = propertyName;
256         this.propertyValueConvertToClass = propertyValueConvertToClass;
257         LOGGER.trace("propertyName:[{}]", propertyName);
258     }

260     /**
261      * The Constructor.
262      * *
263      * <h3>使用场景:</h3>
264      * <blockquote>
265      * 诸如需要排序的对象指定属性类型是数字,但是申明类型的时候由于种种原因是字符串,<br>
266      * 此时需要排序,如果不转成Integer比较的话,字符串比较13 将会在 3数字的前面
267      * </blockquote>
268      *
269      * @param propertyName
270      *            指定bean对象排序属性名字(默认正序).
271      * 
272      *            <p>
273      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
274      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,<br>
275      *            该属性的value 必须实现 {@link Comparable}接口.
276      *            </p>
277      * @param propertyValueConvertToClass
278      *            反射提取出来的值,需要类型转成到的类型
279      * @param comparator
280      *            the comparator
281      * @since 1.5.4
282      */
283     @SuppressWarnings("rawtypes")
284     public PropertyComparator(String propertyName, Class<? extends Comparable> propertyValueConvertToClass, Comparator comparator){
285         Validate.notBlank(propertyName, "propertyName can't be blank!");
286         this.propertyName = propertyName;
287         this.propertyValueConvertToClass = propertyValueConvertToClass;
288         this.comparator = comparator;
289         LOGGER.trace("propertyName:[{}]", propertyName);
290     }

292     /**
293      * Compare.
294      *
295      * @param t1
296      *            the t1
297      * @param t2
298      *            the t2
299      * @return the int
300      * @see org.apache.commons.lang3.ObjectUtils#compare(Comparable, Comparable)
301      * @see org.apache.commons.lang3.ObjectUtils#compare(Comparable, Comparable, boolean)
302      */
303     @SuppressWarnings({ "unchecked", "rawtypes" })
304     @Override
305     public int compare(T t1,T t2){
306         if (t1 == t2){
307             return 0;
308         }else if (null == t1){//null 排在后面
309             return 1;
310         }else if (null == t2){
311             return -1;
312         }
313 
314         Comparable propertyValue1 = PropertyUtil.getProperty(t1, propertyName);
315         Comparable propertyValue2 = PropertyUtil.getProperty(t2, propertyName);
316 
317         //如果值需要类型转换
318         if (null != propertyValueConvertToClass){
319             propertyValue1 = ConvertUtil.convert(propertyValue1, propertyValueConvertToClass);
320             propertyValue2 = ConvertUtil.convert(propertyValue2, propertyValueConvertToClass);
321         }
322         return null != comparator ? comparator.compare(propertyValue1, propertyValue2) : compare(t1, t2, propertyValue1, propertyValue2);
323     }

325     /**
326      * 先比较 propertyValue1以及propertyValue2,再比较 t1/t2 .
327      * 
328      * <p>
329      * 由于我们是提取 property的特殊性, 如果只判断值的话, 那么 TreeSet / TreeMap 过滤掉同sort字段但是对象不相同的情况
330      * </p>
331      *
332      * @param t1
333      *            the t1
334      * @param t2
335      *            the t2
336      * @param propertyValue1
337      *            the property value1
338      * @param propertyValue2
339      *            the property value2
340      * @return the int
341      * @see org.apache.commons.collections4.comparators.ComparableComparator
342      * @since 1.5.4
343      */
344     @SuppressWarnings({ "rawtypes", "unchecked" })
345     private int compare(T t1,T t2,Comparable propertyValue1,Comparable propertyValue2){
346         //NullPointException if propertyValue1.compareTo(propertyValue2)
347         int compareTo = ObjectUtils.compare(propertyValue1, propertyValue2);
348 
349         if (0 == compareTo){
350             //避免TreeSet / TreeMap 过滤掉同sort字段但是对象不相同的情况
351             int hashCode1 = t1.hashCode();
352             int hashCode2 = t2.hashCode();
353             compareTo = ObjectUtils.compare(hashCode1, hashCode2);
354 
355             String pattern = "propertyName:[{}],same propertyValue:[{}],hashCode1:[{}],hashCode2:[{}],result:[{}]";
356             LOGGER.trace(pattern, propertyName, propertyValue1, hashCode1, hashCode2, compareTo);
357             return compareTo;
358         }
359 
360         String pattern = "propertyName:[{}],propertyValue1:[{}],propertyValue2:[{}],result:[{}]";
361         LOGGER.trace(pattern, propertyName, propertyValue1, propertyValue2, compareTo);
362         return compareTo;
363     }
