33  /**
34   * {@link NumberFormat}是所有数值格式的抽象基类,此类提供格式化和解析数值的接口.
35   * 
36   * <p>
37   * 直接已知子类: {@link ChoiceFormat}, {@link DecimalFormat}.<br>
38   * 注意:<span style="color:red">{@link DecimalFormat}不是同步的 </span>,建议为每个线程创建独立的格式实例.(见JAVA API 文档)
39   * </p>
40   * 
41   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
42   * @see Format
43   * @see NumberFormat
44   * @see DecimalFormat
45   * @see NumberPattern
46   * @since 1.0.2
47   */

50      /** The Constant LOGGER. */
51      private static final Logger LOGGER = LoggerFactory.getLogger(NumberFormatUtil.class);

53      /** Don't let anyone instantiate this class. */
54      private NumberFormatUtil(){
55          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
56          //see 《Effective Java》 2nd
57          throw new AssertionError("No " + getClass().getName() + " instances for you!");
58      }
59  
60      /**
61       * 将 {@link Number} 使用 {@link RoundingMode} <code>numberPattern</code>格式化.
62       * 
63       * <h3>关于参数 <code>value</code>:</h3>
64       * 
65       * <blockquote>
66       * <p>
67       * <b>请尽量传递Integer,Long,BigDecimal,而不要使用 float,double等浮点类型</b>,否则可能结果不准确,特别是jdk8以下的版本,具体参见
68       * <a href="https://github.com/venusdrogon/feilong-core/issues/165">NumberFormatUtilTest 在 jdk8 下面测试不通过</a>
69       * </p>
70       * 
71       * </blockquote>
72       * 
73       * <h3>关于参数 <code>roundingMode</code></h3>
74       * 
75       * <blockquote>
76       * <p>
77       * 虽然{@link DecimalFormat},默认使用的是银行家舍入法 {@link RoundingMode#HALF_EVEN},参见
78       * <a href="../util/NumberUtil.html#RoundingMode_HALF_EVEN">关于 RoundingMode#HALF_EVEN</a>
79       * </p>
80       * 
81       * <p>
82       * 但是我们常用标准的四舍五入,为了保持和 {@link com.feilong.core.lang.NumberUtil#toString(Number, String)}一致性,如果 roundingMode 为null,使用常用的
83       * {@link RoundingMode#HALF_UP}来进行格式化
84       * </p>
85       * </blockquote>
86       * 
87       * <h3>示例:</h3>
88       * 
89       * <blockquote>
90       * 
91       * <pre class="code">
92       * NumberFormatUtil.format(toBigDecimal(1.15), "#####.#",null)     =   1.2
93       * NumberFormatUtil.format(toBigDecimal(1.25), "#####.#",null)     =   1.3
94       * NumberFormatUtil.format(toBigDecimal(1.251), "#####.#",null)    =   1.3
95       * 
96       * NumberFormatUtil.format(toBigDecimal(-1.15), "#####.#",null)    =   -1.2
97       * NumberFormatUtil.format(toBigDecimal(-1.25), "#####.#",null)    =   -1.3
98       * NumberFormatUtil.format(toBigDecimal(-1.251), "#####.#",null)   =   -1.3
99       * 
100      * 
101      * NumberFormatUtil.format(toBigDecimal(1.15), "#####.#", RoundingMode.HALF_EVEN)     =   1.2
102      * NumberFormatUtil.format(toBigDecimal(1.25), "#####.#", RoundingMode.HALF_EVEN)     =   1.2
103      * NumberFormatUtil.format(toBigDecimal(1.251), "#####.#", RoundingMode.HALF_EVEN)    =   1.3
104      * 
105      * NumberFormatUtil.format(toBigDecimal(-1.15), "#####.#", RoundingMode.HALF_EVEN)    =   -1.2
106      * NumberFormatUtil.format(toBigDecimal(-1.25), "#####.#", RoundingMode.HALF_EVEN)    =   -1.2
107      * NumberFormatUtil.format(toBigDecimal(-1.251), "#####.#", RoundingMode.HALF_EVEN)   =   -1.3
108      * </pre>
109      * 
110      * </blockquote>
111      *
112      * @param value
113      *            the value
114      * @param numberPattern
115      *            格式化数字格式,可以参见或者使用{@link NumberPattern}
116      * @param roundingMode
117      *            舍入模式{@link RoundingMode},如果 为null,使用常用的 {@link RoundingMode#HALF_UP}
118      * @return 如果 <code>value</code> 是null,抛出 {@link NullPointerException}<br>
119      *         如果 <code>numberPattern</code> 是null,抛出 {@link NullPointerException}<br>
120      *         如果 <code>numberPattern</code> 是blank,抛出 {@link IllegalArgumentException}<br>
121      * @see DecimalFormat
122      * @see <a href="../util/NumberUtil.html#RoundingMode">JAVA 8种舍入法</a>
123      */
124     public static String format(Number value,String numberPattern,RoundingMode roundingMode){
125         Validate.notNull(value, "value can't be null!");
126         Validate.notBlank(numberPattern, "numberPattern can't be null!");
127 
128         //该构造方法内部 调用了applyPattern(pattern, false)
129         DecimalFormat decimalFormat = new DecimalFormat(numberPattern);
130         decimalFormat.setRoundingMode(defaultIfNull(roundingMode, HALF_UP));
131 
132         String result = decimalFormat.format(value);
133         LOGGER.trace("input:[{}],with:[{}]=[{}],localizedPattern:[{}]", value, numberPattern, result, decimalFormat.toLocalizedPattern());
134         return result;
135     }
136 }
