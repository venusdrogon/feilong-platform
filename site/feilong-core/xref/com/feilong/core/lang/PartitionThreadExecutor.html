16  package com.feilong.core.lang;
17  
18  import java.util.List;
19  import java.util.Map;
20  
21  /**
22   * 分区线程执行器.
23   * 
24   * <h3>背景:</h3>
25   * 
26   * <blockquote>
27   * 
28   * <p>
29   * 经常有如此的需求, 下载客户8000张图片加工处理, 如果串行执行耗时很长,此时需要使用多线程,<br>
30   * 但是如果从头开始写会开发蛮久时间,此时把核心功能进行封装 ,你只需要传入 8000和url list, 告知每个线程处理多少张图片, 以及每个线程的代码处理方式即可,实现类会自动分区创建线程并执行
31   * </p>
32   * 
33   * <p>
34   * 契合 feilong 的思想, Reduce development, Release ideas (减少开发,释放思想)
35   * </p>
36   * </blockquote>
37   * 
38   * <h3>快速调用以及实现类:</h3>
39   * 
40   * <blockquote>
41   * <p>
42   * 目前已知快速调用有 {@link com.feilong.core.lang.ThreadUtil#execute(List, int, Map, PartitionRunnableBuilder) }, 或者调用
43   * {@link com.feilong.core.lang#DefaultPartitionThreadExecutor}, 或者调用
44   * {@link "com.feilong.spring.scheduling.concurrent.AsyncTaskExecutorPartitionThreadExecutor"}
45   * </p>
46   * 
47   * </blockquote>
48   * 
49   * <h3>如果使用spring asyncTaskExecutor:</h3>
50   * <blockquote>
51   * 
52   * <pre>
53  {@code
54   <!-- 配置线程池 -->
55      <bean id="taskExecutor" class="org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor">
56           <!-- 线程池维护线程的最少数量 -->
57          <property name="corePoolSize" value="10" />
58           <!-- 线程池维护线程所允许的空闲时间 -->
59          <property name="keepAliveSeconds" value="30000" />
60           <!-- 线程池维护线程的最大数量 -->
61          <property name="maxPoolSize" value="1000" />
62           <!-- 线程池所使用的缓冲队列 -->
63          <property name="queueCapacity" value="200" />
64      </bean>
65      
66      <!-- partitionThreadExecutor -->
67  
68      <bean id="partitionThreadExecutor" class="com.feilong.spring.scheduling.concurrent.AsyncTaskExecutorPartitionThreadExecutor">
69          <property name="asyncTaskExecutor" ref="taskExecutor"></property>
70      </bean>
71      
72      }
73   * </pre>
74   * 
75   * 然后 java 代码调用
76   * 
77   * <pre>
78   * 
79   * <code>@Autowired</code>
80   * private PartitionThreadExecutor partitionThreadExecutor;
81   * 
82   * <code>@Test</code>
83   * public void test(){
84   *     final List{@code <Integer>} list = toList(1, 2, 3, 4, 5);
85   * 
86   *     partitionThreadExecutor.excute(list, 2, new PartitionRunnableBuilder{@code <Integer>}(){
87   * 
88   *         public Runnable build(List{@code <Integer>} perBatchList,final PartitionThreadEntity partitionThreadEntity,Map{@code <String, ?>} paramsMap){
89   *             return new Runnable(){
90   * 
91   *                 public void run(){
92   *                     for (Integer integer : list){
93   *                         LOGGER.debug("{}-{}", partitionThreadEntity.getBatchNumber(), integer);
94   *                     }
95   *                 }
96   *             };
97   *         }
98   *     });
99   * }
100  * </pre>
101  * 
102  * </blockquote>
103  *
104  * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
105  * @since 1.11.0
106  */
107 public interface PartitionThreadExecutor{
108 
109     /**
110      * 给定一个待解析的 <code>list</code>,设定每个线程执行多少条 <code>eachSize</code> ,使用自定义的
111      * <code>partitionRunnableBuilder</code>,自动<span style="color:green">构造多条线程</span>并运行.
112      * 
113      * <p>
114      * 如果 <code>list</code> 是null,抛出 {@link NullPointerException}<br>
115      * 如果 <code>list</code> 是empty,抛出 {@link IllegalArgumentException}<br>
116      * 如果 {@code eachSize <=0} ,抛出 {@link IllegalArgumentException}<br>
117      * 如果 <code>partitionRunnableBuilder</code> 是null,抛出 {@link NullPointerException}<br>
118      * </p>
119      * 
120      * @param <T>
121      *            the generic type
122      * @param list
123      *            执行解析的list
124      * 
125      *            <p>
126      *            比如 100000个 User,不能为null或者empty
127      *            </p>
128      * @param eachSize
129      *            每个线程执行多少个对象
130      * 
131      *            <p>
132      *            比如 一个线程解析 1000个 User, 那么程序内部 会自动创建 100000/1000个 线程去解析;<br>
133      *            必须{@code >}0
134      *            </p>
135      * @param partitionRunnableBuilder
136      *            每个线程做的事情,不能为null
137      * @see com.feilong.core.lang.ThreadUtil#execute(List, int, Map, PartitionRunnableBuilder)
138      */
139     <T> void excute(List<T> list,int eachSize,PartitionRunnableBuilder<T> partitionRunnableBuilder);
140 
141     //---------------------------------------------------------------
142 
143     /**
144      * 给定一个待解析的 <code>list</code>,设定每个线程执行多少条 <code>eachSize</code>,传入一些额外的参数 <code>paramsMap</code>,使用自定义的
145      * <code>partitionRunnableBuilder</code>,自动<span style="color:green">构造多条线程</span>并运行.
146      * 
147      * <p>
148      * 如果 <code>list</code> 是null,抛出 {@link NullPointerException}<br>
149      * 如果 <code>list</code> 是empty,抛出 {@link IllegalArgumentException}<br>
150      * 如果 {@code eachSize <=0} ,抛出 {@link IllegalArgumentException}<br>
151      * 如果 <code>partitionRunnableBuilder</code> 是null,抛出 {@link NullPointerException}<br>
152      * </p>
153      * 
154      * @param <T>
155      *            the generic type
156      * @param list
157      *            执行解析的list
158      * 
159      *            <p>
160      *            比如 100000个 User,不能为null或者empty
161      *            </p>
162      * @param eachSize
163      *            每个线程执行多少个对象
164      * 
165      *            <p>
166      *            比如 一个线程解析 1000个 User, 那么程序内部 会自动创建 100000/1000个 线程去解析;<br>
167      *            必须{@code >}0
168      *            </p>
169      * @param paramsMap
170      *            自定义的相关参数
171      * 
172      *            <p>
173      *            该参数目的是你可以在自定义的 <code>partitionRunnableBuilder</code>中使用;<br>
174      *            如果你传入的<code>partitionRunnableBuilder</code>中不需要额外的自定义参数,那么此处可以传入null
175      *            </p>
176      * @param partitionRunnableBuilder
177      *            每个线程做的事情,不能为null
178      * @see com.feilong.core.lang.ThreadUtil#execute(List, int, Map, PartitionRunnableBuilder)
179      */
180     <T> void excute(List<T> list,int eachSize,Map<String, ?> paramsMap,PartitionRunnableBuilder<T> partitionRunnableBuilder);
181 }
