41  /**
42   * focus on {@link Field} 反射工具类.
43   * 
44   * <h3>{@link Field} 相关的几个方法的区别:</h3>
45   * 
46   * <blockquote>
47   * <table border="1" cellspacing="0" cellpadding="4" summary="">
48   * 
49   * <tr style="background-color:#ccccff">
50   * <th align="left">字段</th>
51   * <th align="left">说明</th>
52   * </tr>
53   * 
54   * <tr valign="top">
55   * <td>{@link Class#getDeclaredFields() getDeclaredFields}</td>
56   * <td>返回 {@link Field} 对象的一个数组,这些对象反映此 Class 对象所表示的类或接口所声明的<span style="color:green">所有字段</span>.
57   * 
58   * <p>
59   * 包括<span style="color:green"><b>公共</b>、<b>保护</b>、<b>默认(包)</b>访问和<b>私有</b>字段</span>, <span style="color:red">但不包括继承的字段</span>.
60   * </p>
61   * 
62   * <b>返回数组中的元素没有排序,也没有任何特定的顺序.</b><br>
63   * 如果该类或接口不声明任何字段,或者此 Class 对象表示一个基本类型、一个数组类或 void,则此方法返回一个长度为 0 的数组.</td>
64   * </tr>
65   * 
66   * <tr valign="top" style="background-color:#eeeeff">
67   * <td>{@link Class#getFields() getFields}</td>
68   * <td>返回一个包含某些 {@link Field} 对象的数组,这些对象反映此 Class 对象所表示的类或接口的所有 <span style="color:red">可访问公共字段</span>.
69   * 
70   * <p>
71   * 特别地,如果该 Class 对象表示<b>一个类</b>,则此方法返回<span style="color:red">该类及其所有超类的公共字段</span>.<br>
72   * 如果该 Class 对象表示<b>一个接口</b>,则此方法返回<span style="color:red">该接口及其所有超接口的公共字段</span>.
73   * </p>
74   * 
75   * <p>
76   * 返回数组中的元素没有排序,也没有任何特定的顺序.
77   * </p>
78   * <br>
79   * 
80   * 如果类或接口没有可访问的公共字段,或者表示一个数组类、一个基本类型或 void,则此方法返回长度为 0 的数组. <br>
81   * 该方法不反映数组类的隐式长度字段.用户代码应使用 {@link java.lang.reflect.Array} 类的方法来操作数组.</td>
82   * </tr>
83   * 
84   * <tr valign="top">
85   * <td>{@link Class#getDeclaredField(String) getDeclaredField}</td>
86   * <td>返回一个 {@link Field} 对象,该对象反映此 Class 对象所表示的类或接口的指定已声明字段.<br>
87   * 注意,此方法不反映数组类的 length 字段.</td>
88   * </tr>
89   * 
90   * <tr valign="top" style="background-color:#eeeeff">
91   * <td>{@link Class#getField(String) getField}</td>
92   * <td>返回一个 {@link Field} 对象,它反映此 Class 对象所表示的类或接口的指定<span style="color:green">公共成员字段</span>.<br>
93   * 要反映的字段由下面的算法确定.<br>
94   * 设 C 为此对象所表示的类: <br>
95   * 如果 C 声明一个带有指定名的公共字段,则它就是要反映的字段. <br>
96   * 如果在第 1 步中没有找到任何字段,则该算法被递归地应用于 C 的每一个<b>直接超接口</b>.直接超接口按其声明顺序进行搜索. <br>
97   * 如果在第 1、2 两步没有找到任何字段,且 C 有一个<b>超类 S</b>,则在 S 上递归调用该算法.如果 C 没有超类,则抛出 NoSuchFieldException.<br>
98   * </td>
99   * </tr>
100  * 
101  * </table>
102  * </blockquote>
103  * 
104  * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
105  * @see org.apache.commons.lang3.reflect.FieldUtils
106  * @see "org.springframework.util.ReflectionUtils"
107  * @since 1.0.7
108  */
109 public final class FieldUtil{
110 
111     /** The Constant LOGGER. */
112     private static final Logger LOGGER = LoggerFactory.getLogger(FieldUtil.class);
113 
114     /** Don't let anyone instantiate this class. */
115     private FieldUtil(){
116         //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
117         //see 《Effective Java》 2nd
118         throw new AssertionError("No " + getClass().getName() + " instances for you!");
119     }

123     /**
124      * 获得对象 <code>obj</code> 所有字段的值(<b>不是属性</b>),key是 fieldName,value 是值.
125      * 
126      * <h3>说明:</h3>
127      * <blockquote>
128      * <ol>
129      * <li>是所有字段的值(<b>不是属性</b>)</li>
130      * <li>自动过滤私有并且静态的字段,比如 LOGGER serialVersionUID</li>
131      * <li>返回的map 是 TreeMap,顺序是 field name 的顺序</li>
132      * </ol>
133      * </blockquote>
134      * 
135      * <h3>示例:</h3>
136      * 
137      * <blockquote>
138      * 
139      * <pre class="code">
140      * User user = new User(12L);
141      * LOGGER.debug(JsonUtil.format(FieldUtil.getAllFieldNameAndValueMap(user, "date")));
142      * </pre>
143      * 
144      * <b>返回:</b>
145      * 
146      * <pre class="code">
147      * {
148      * "age": null,
149      * "id": 12
150      * }
151      * </pre>
152      * 
153      * </blockquote>
154      *
155      * @param obj
156      *            the obj
157      * @param excludeFieldNames
158      *            需要排除的field names,如果是null或者empty, 那么不会判断
159      * @return 如果 <code>obj</code> 是null,抛出 {@link NullPointerException}<br>
160      *         如果 <code>excludeFieldNames</code> 是null或者empty,解析所有的field<br>
161      *         如果 <code>obj</code>没有字段或者字段都被参数 <code>excludeFieldNames</code> 排除掉了,返回 {@link Collections#emptyMap()}<br>
162      * @see #getAllFieldList(Class, String...)
163      * @see #getFieldValue(Object, String)
164      */
165     public static Map<String, Object> getAllFieldNameAndValueMap(Object obj,String...excludeFieldNames){
166         Validate.notNull(obj, "obj can't be null!");
167 
168         List<Field> fieldList = getAllFieldList(obj.getClass(), excludeFieldNames);
169         if (isNullOrEmpty(fieldList)){
170             return emptyMap();
171         }
172 
173         //---------------------------------------------------------------
174         Map<String, Object> map = new TreeMap<>();
175         for (Field field : fieldList){
176             map.put(field.getName(), getFieldValue(obj, field.getName()));
177         }
178         return map;
179     }

183     /**
184      * 获得 <code>klass</code> 排除某些 <code>excludeFieldNames</code> 之后的字段list.
185      * 
186      * <h3>说明:</h3>
187      * <blockquote>
188      * <ol>
189      * <li>是所有字段的值(<b>不是属性</b>)</li>
190      * <li>自动过滤私有并且静态的字段,比如 LOGGER serialVersionUID</li>
191      * </ol>
192      * </blockquote>
193      * 
194      * @param klass
195      *            the klass
196      * @param excludeFieldNames
197      *            需要排除的field names,如果传递过来是null或者empty 那么不会判断
198      * @return 如果 <code>obj</code> 是null,抛出 {@link NullPointerException}<br>
199      *         如果 <code>excludeFieldNames</code> 是null或者empty,解析所有的field<br>
200      *         如果 {@link FieldUtils#getAllFieldsList(Class)} 是null或者empty,返回 {@link Collections#emptyList()}<br>
201      *         如果 <code>obj</code>没有字段或者字段都被参数 <code>excludeFieldNames</code> 排除掉了,返回 {@link Collections#emptyMap()}<br>
202      * @see FieldUtils#getAllFieldsList(Class)
203      * @since 1.7.1
204      */
205     public static List<Field> getAllFieldList(final Class<?> klass,String...excludeFieldNames){
206         Validate.notNull(klass, "klass can't be null!");
207         //---------------------------------------------------------------
208 
209         //获得给定类的所有声明字段 {@link Field},包括所有的parents,包括 public/protect/private/inherited...
210         List<Field> fieldList = FieldUtils.getAllFieldsList(klass);
211         if (isNullOrEmpty(fieldList)){
212             return emptyList();
213         }
214 
215         //---------------------------------------------------------------
216         Predicate<Field> excludeFieldPredicate = BeanPredicateUtil.containsPredicate("name", excludeFieldNames);
217         Predicate<Field> staticPredicate = new Predicate<Field>(){
218 
219             @Override
220             public boolean evaluate(Field field){
221                 int modifiers = field.getModifiers();
222                 // 私有并且静态 一般是log 或者  serialVersionUID
223                 boolean isStatic = Modifier.isStatic(modifiers);
224 
225                 String pattern = "[{}.{}],modifiers:[{}]{}";
226                 LOGGER.trace(pattern, klass.getSimpleName(), field.getName(), modifiers, isStatic ? " [isStatic]" : EMPTY);
227                 return isStatic;
228             }
229         };
230         return selectRejected(fieldList, PredicateUtils.orPredicate(excludeFieldPredicate, staticPredicate));
231     }

235     /**
236      * 得到某个对象 <code>owner</code> 的公共字段 <code>fieldName</code> 值.
237      *
238      * @param <T>
239      *            the generic type
240      * @param obj
241      *            the owner
242      * @param fieldName
243      *            the field name
244      * @return 如果 <code>owner</code> 是null,抛出 {@link NullPointerException}<br>
245      *         如果 <code>fieldName</code> 是null,抛出 {@link NullPointerException}<br>
246      *         如果 <code>fieldName</code> 是blank,抛出 {@link IllegalArgumentException}<br>
247      * @see org.apache.commons.lang3.reflect.FieldUtils#readField(Object, String, boolean)
248      * @since 1.4.0
249      * @since 1.9.2 change to private
250      */
251     @SuppressWarnings("unchecked")
252     private static <T> T getFieldValue(Object obj,String fieldName){
253         try{
254             return (T) FieldUtils.readField(obj, fieldName, true);
255         }catch (IllegalAccessException e){
256             String message = Slf4jUtil.format("ownerClass:[{}],fieldName:[{}],ownerObj:[{}]", obj.getClass().getName(), fieldName, obj);
257             LOGGER.error(message, e);
258             throw new ReflectException(message, e);
259         }
260     }
