16  package com.feilong.core.lang;

18  import java.lang.reflect.Array;
19  
20  import org.apache.commons.lang3.Validate;
21  
22  /**
23   * 数组工具类.
24   * 
25   * <h3>提供以下方法:</h3>
26   * <blockquote>
27   * <ol>
28   * <li>{@link #getElement(Object, int)},得到数组中的某个元素</li>
29   * <li>{@link #newArray(Class, int)},构造一个指定类型以及指定长度的数组.</li>
30   * </ol>
31   * </blockquote>
32   * 
33   * 
34   * <h3>判断是否包含:</h3>
35   * 
36   * <blockquote>
37   * <ul>
38   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(boolean[], boolean)}</li>
39   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(byte[], byte)}</li>
40   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(char[], char)}</li>
41   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(double[], double)}</li>
42   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(float[], float)}</li>
43   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(int[], int)}</li>
44   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(long[], long)}</li>
45   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(Object[], Object)}</li>
46   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(short[], short)}</li>
47   * <li>{@link org.apache.commons.lang3.ArrayUtils#contains(double[], double, double)}</li>
48   * </ul>
49   * </blockquote>
50   * 
51   * <h3>判断对象是数组:</h3>
52   * <blockquote>
53   * <ol>
54   * <li>如果你想detect所有的 array类型,你必须判断一个object is an instanceof boolean[], byte[], short[], char[], int[], long[], float[], double[], or
55   * Object[]</li>
56   * <li>Object[] 数组 Integer/String...自定义的对象User.等数组也 instanceof Object[],二维数组不管是primitive 还是包装类型,都instanceof Object[];<br>
57   * so depending on how you want to handle nested arrays, it can get complicated.</li>
58   * </ol>
59   * </blockquote>
60   *
61   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
62   * @see org.apache.commons.lang3.ArrayUtils
63   * @since 1.4.0
64   */
65  public final class ArrayUtil{
66  
67      /** Don't let anyone instantiate this class. */
68      private ArrayUtil(){
69          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
70          //see 《Effective Java》 2nd
71          throw new AssertionError("No " + getClass().getName() + " instances for you!");
72      }
73  
74      //---------------------------------------------------------------
75  
76      /**
77       * 得到指定数组对象 <code>array</code> 中指定索引 <code>index</code> 的元素.
78       * 
79       * <h3>说明:</h3>
80       * <blockquote>
81       * <ol>
82       * <li>支持 primitive type类型数组</li>
83       * <li>
84       * Returns the value of the indexed component in the specified array object. <br>
85       * The value is automatically wrapped in an object if it has a primitive type.
86       * </li>
87       * </ol>
88       * </blockquote>
89       * 
90       * <h3>示例:</h3>
91       * 
92       * <blockquote>
93       * 
94       * <pre class="code">
95       * ArrayUtil.getElement(new String[] { "jinxin", "feilong", "1" }, 2)   =   1
96       * ArrayUtil.getElement(new int[] { 5, 8, 2, 0 }, 2)                    =   2
97       * </pre>
98       * 
99       * </blockquote>
100      *
101      * @param <T>
102      *            the generic type
103      * @param array
104      *            数组
105      * @param index
106      *            索引,从0开始
107      * @return 如果指定的参数{@code index}是负数,或者大于等于指定数组 <code>array</code> 的长度,抛出 {@link ArrayIndexOutOfBoundsException}
108      * @throws NullPointerException
109      *             如果 <code>array</code> 是null
110      * @throws IllegalArgumentException
111      *             如果 <code>array</code> 不是数组
112      * @see java.lang.reflect.Array#get(Object, int)
113      */
114     @SuppressWarnings("unchecked")
115     public static <T> T getElement(Object array,int index){
116         Validate.notNull(array, "array can't be null!");
117         return (T) Array.get(array, index);
118     }
119 
120     /**
121      * 构造一个指定类型 <code>componentType</code> 以及指定长度 <code>length</code>的数组.
122      * 
123      * <h3>示例:</h3>
124      * 
125      * <blockquote>
126      * 
127      * <pre class="code">
128      * assertArrayEquals(new Integer[] {}, ArrayUtil.newArray(Integer.class, 0));
129      * assertArrayEquals(new Integer[] { null, null, null }, ArrayUtil.newArray(Integer.class, 3));
130      * </pre>
131      * 
132      * </blockquote>
133      *
134      * @param <T>
135      *            the generic type
136      * @param componentType
137      *            数组的类型
138      * @param length
139      *            数组的长度
140      * @return 如果 <code>componentType</code> 是null,抛出 {@link NullPointerException}<br>
141      *         如果 {@code length < 0} ,抛出 {@link IllegalArgumentException}<br>
142      * @see java.lang.reflect.Array#newInstance(Class, int)
143      * @see java.lang.reflect.Array#newInstance(Class, int...)
144      * @see "com.google.common.collect#newArray(Class, int)"
145      * @since 1.6.1
146      */
147     @SuppressWarnings("unchecked")
148     public static <T> T[] newArray(Class<T> componentType,int length){
149         Validate.notNull(componentType, "componentType can't be null!");
150         Validate.isTrue(length >= 0, "length:[%s],must >=0", length);
151         return (T[]) Array.newInstance(componentType, length);
152     }
