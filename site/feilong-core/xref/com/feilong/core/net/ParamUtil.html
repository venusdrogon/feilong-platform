73      /**
74       * 给指定的<code>uriString</code>添加指定的参数 <code>paramName</code> 和值 <code>paramValue</code>.
75       * 
76       * <h3>说明:</h3>
77       * <blockquote>
78       * <ol>
79       * <li>如果原来的<code>uriString</code>没有指定的参数<code>paramName</code>,那么追加新的参数<code>paramName</code>和值<code>paramValue</code>.</li>
80       * <li>如果原来的<code>uriString</code>有指定的参数<code>paramName</code>,那么会被新的值替换<code>paramValue</code>.</li>
81       * <li>如果原来的<code>uriString</code>有指定的参数<code>paramName</code>,并且是多值类型(参数数组),那么多值参数中第一个会被新的值替换<code>paramValue</code>,其他的被丢弃.</li>
82       * <li>如果原来的<code>uriString</code>有参数,不管是拼接还是替换都会保持参数的原始顺序.</li>
83       * <li>如果<code>uriString</code>带有? 和参数,会先被截取,最后再拼接.</li>
84       * <li>如果<code>uriString</code>不带?,则自动增加?</li>
85       * </ol>
86       * </blockquote>
87       * 
88       * <h3>示例:</h3>
89       * <blockquote>
90       * 
91       * <pre class="code">
92       * String uri = "http://www.feilong.com:8888/esprit-frontend/search.htm?{@code keyword=%E6%81%A4&page=}";
93       * String pageParamName = "label";
94       * String prePageNo = "2-5-8-12";
95       * LOGGER.info(ParamUtil.addParameter(uri, pageParamName, prePageNo, UTF8));
96       * </pre>
97       * 
98       * <b>返回:</b>
99       * 
100      * <pre class="code">
101      * {@code http://www.feilong.com:8888/esprit-frontend/search.htm?keyword=%E6%81%A4&page=&label=2-5-8-12}
102      * </pre>
103      * 
104      * </blockquote>
105      *
106      * @param uriString
107      *            如果带有? 和参数,会先被截取,最后再拼接,<br>
108      *            如果不带?,则自动增加?
109      * @param paramName
110      *            添加的参数名称
111      * @param paramValue
112      *            添加的参数值
113      * @param charsetType
114      *            字符编码,建议使用 {@link CharsetType} 定义好的常量<br>
115      *            <span style="color:green">如果是null或者 empty,那么参数部分原样返回,自行处理兼容性问题</span><br>
116      *            否则会先解码,再加码,因为ie浏览器和chrome浏览器 url中访问路径 ,带有中文情况下不一致
117      * @return 如果 <code>uriString</code> 是null或者empty,返回 {@link StringUtils#EMPTY}<br>
118      *         如果 <code>paramName</code> 是null,以 {@link StringUtils#EMPTY}替代<br>
119      *         如果 <code>paramValue</code> 是null,以 {@link StringUtils#EMPTY}替代<br>
120      * @see #addParameterSingleValueMap(String, Map, String)
121      * @since 1.9.0 change paramValue type from Object to String
122      */
123     public static String addParameter(String uriString,String paramName,String paramValue,String charsetType){
124         return addParameterSingleValueMap(uriString, toMap(paramName, paramValue), charsetType);
125     }

184      * @param uriString

186      *            如果不带?,则自动增加?
187      * @param singleValueMap

189      * @param charsetType

328     /**
329      * 将{@code a=1&b=2}这样格式的数据转换成map (如果<code>charsetType</code> 不是null或者empty 返回安全的 key和value).
330      * 
331      * <h3>说明:</h3>
332      * <blockquote>
333      * <ol>
334      * <li>内部使用 {@link LinkedHashMap},map顺序依照 <code>queryString</code> 逗号分隔的顺序</li>
335      * <li>解析方式:参数和参数之间是以 {@code &} 分隔, 参数的key和value 是以 = 号分隔</li>
336      * </ol>
337      * </blockquote>
338      * 
339      * <h3>示例1:</h3>
340      * <blockquote>
341      * 
342      * <pre class="code">
343      * LOGGER.info(JsonUtil.format(ParamUtil.toSafeArrayValueMap("{@code a=1&b=2&a=5}", UTF8)));
344      * </pre>
345      * 
346      * <b>返回:</b>
347      * 
348      * <pre class="code">
349      {"a": [
350                  "1",
351                  "5"
352              ],
353              "b": ["2"]
354       }
355      * </pre>
356      * 
357      * <hr>
358      * 
359      * <pre class="code">
360      * LOGGER.info(JsonUtil.format(ParamUtil.toSafeArrayValueMap("{@code a=&b=2&a}", UTF8)));
361      * </pre>
362      * 
363      * <b>返回:</b>
364      * 
365      * <pre class="code">
366      {"a": [
367              "",
368              ""
369          ],
370       "b": ["2"]
371      }
372      * </pre>
373      * 
374      * </blockquote>
375      *
376      * @param queryString
377      *            {@code a=1&b=2}类型的数据,支持{@code a=1&a=1}的形式, 返回map的值是数组
378      * @param charsetType
379      *            字符编码,建议使用 {@link CharsetType} 定义好的常量<br>
380      *            <span style="color:green">如果是null或者 empty,那么参数部分原样返回,自行处理兼容性问题</span><br>
381      *            否则会先解码,再加码,因为ie浏览器和chrome浏览器 url中访问路径 ,带有中文情况下不一致
382      * @return 如果 <code>queryString</code> 是null或者empty,返回 {@link Collections#emptyMap()}<br>
383      * @see org.apache.commons.lang3.ArrayUtils#add(Object[], Object)
384      * @see com.feilong.core.lang.StringUtil#split(String, String)
385      * @since 1.4.0
386      */
387     static Map<String, String[]> toSafeArrayValueMap(String queryString,String charsetType){
388         if (isNullOrEmpty(queryString)){
389             return emptyMap();
390         }
391 
392         String[] nameAndValueArray = StringUtil.split(queryString, AMPERSAND);
393         int length = nameAndValueArray.length;
394 
395         Map<String, String[]> safeArrayValueMap = newLinkedHashMap(length);//使用 LinkedHashMap 保证元素的顺序
396         for (int i = 0; i < length; ++i){
397             String[] tempArray = nameAndValueArray[i].split("=", 2);
398 
399             String key = decodeAndEncode(tempArray[0], charsetType);
400             String value = tempArray.length == 2 ? tempArray[1] : EMPTY;//有可能参数中,只有名字没有值或者值是空,处理的时候不能遗失掉
401             value = decodeAndEncode(value, charsetType);
402 
403             safeArrayValueMap.put(key, ArrayUtils.add(safeArrayValueMap.get(key), value));
404         }
405         return safeArrayValueMap;
406     }

410     /**
411      * 将 <code>singleValueMap</code> 转成<code>自然排序</code>, 然后将key和value直接拼接成字符串(<span style="color:red">不使用 = 和& 分隔</span>).
412      * 
413      * <p>
414      * 已知,适用于 网易易盾验证码
415      * </p>
416      * 
417      * <h3>示例:</h3>
418      * <blockquote>
419      * 
420      * <pre class="code">
421      * Map{@code <String, String>} map = new HashMap{@code <>}();
422      * map.put("service", "create_salesorder");
423      * map.put("_input_charset", "gbk");
424      * map.put("totalActual", "210.00");
425      * map.put("address", "江苏南通市通州区888组888号");
426      * LOGGER.debug(ParamUtil.toNaturalOrderingKeyJoinValue(map));
427      * </pre>
428      * 
429      * <b>返回:</b>
430      * 
431      * <pre class="code">
432      * {@code _input_charsetgbkaddress江苏南通市通州区888组888号servicecreate_salesordertotalActual210.00}
433      * </pre>
434      * 
435      * </blockquote>
436      * 
437      * <h3>规则:</h3>
438      * 
439      * <blockquote>
440      * 
441      * <ol>
442      * <li>首先将<code>singleValueMap</code> 使用 {@link SortUtil#sortMapByKeyAsc(Map)} 进行排序,</li>
443      * <li>然后将map的key和value 直接连接</li>
444      * </ol>
445      * 
446      * </blockquote>
447      * 
448      * <h3>说明:</h3>
449      * <blockquote>
450      * <ol>
451      *

474      * </pre>

477      *

479      * {@code create_salesorderprovince江苏省totalActual}
480      * </pre>
481      * 
482      * </blockquote>
483      * </li>
484      * 
485      * </ol>
486      * </blockquote>
487      * 
488      * @param singleValueMap
489      *            用于拼接签名的参数
490      * @return 如果 <code>singleValueMap</code> 是null或者empty,返回 {@link StringUtils#EMPTY}<br>
491      *         否则将<code>singleValueMap</code>排序之后,循环直接拼接key和value
492      * @since 1.10.6

494     public static String toNaturalOrderingKeyJoinValue(Map<String, String> singleValueMap){
495         if (isNullOrEmpty(singleValueMap)){
496             return EMPTY;
497         }
498 
499         //---------------------------------------------------------------
500         Map<String, String> sortMapByKeyAsc = sortMapByKeyAsc(singleValueMap);
501 
502         StringBuilder sb = new StringBuilder();
503         for (Map.Entry<String, String> entry : sortMapByKeyAsc.entrySet()){//有顺序的参数

505             String value = entry.getValue();

507             sb.append(defaultString(key));
508             sb.append(defaultString(value));
509         }
510         return sb.toString();
511     }

742      * <blockquote>
743      * <ol>

756      * @see "org.springframework.web.servlet.view.RedirectView#appendQueryProperties(StringBuilder,Map, String)"

758      */

769             sb.append(AMPERSAND);//放心大胆的拼接 &, 不判断是否是最后一个,最后会截取
770         }
771 
772         return StringUtil.substringWithoutLast(sb, AMPERSAND);
773     }

863     /**
864      * To safe array value map.
865      * 
866      * <p>
867      * 内部使用 {@link LinkedHashMap},保持map元素顺序
868      * </p>
869      *
870      * @param arrayValueMap
871      *            the array value map
872      * @param charsetType
873      *            字符编码,建议使用 {@link CharsetType} 定义好的常量<br>
874      *            <span style="color:green">如果是null或者 empty,那么参数部分原样返回,自行处理兼容性问题</span><br>
875      *            否则会先解码,再加码,因为ie浏览器和chrome浏览器 url中访问路径 ,带有中文情况下不一致
876      * @return 如果 <code>arrayValueMap</code> 是null或者empty,返回 {@link Collections#emptyMap()}<br>
877      */
878     private static Map<String, String[]> toSafeArrayValueMap(Map<String, String[]> arrayValueMap,String charsetType){
879         if (isNullOrEmpty(arrayValueMap)){
880             return emptyMap();
881         }
882         Map<String, String[]> safeArrayValueMap = newLinkedHashMap(arrayValueMap.size()); //使用 LinkedHashMap,保持map元素顺序
883         for (Map.Entry<String, String[]> entry : arrayValueMap.entrySet()){
884             String key = entry.getKey();
885             String[] paramValues = entry.getValue();
886             if (isNullOrEmpty(paramValues)){
887                 LOGGER.warn("the param key:[{}] value is null", key);
888                 paramValues = EMPTY_STRING_ARRAY;//赋予 empty数组,为了下面的转换
889             }
890             safeArrayValueMap.put(decodeAndEncode(key, charsetType), toSafeValueArray(paramValues, charsetType));
891         }
892         return safeArrayValueMap;
893     }

916     /**
917      * 浏览器传递queryString()参数差别;
918      * 
919      * <h3>说明:</h3>
920      * <blockquote>
921      * <ol>
922      * <li>主要是为了处理浏览器兼容问题,参见
923      * <a href="http://stackoverflow.com/questions/15004593/java-request-getquerystring-value-different-between-chrome-and-ie-browser">
924      * java-request-getquerystring-value-different-between-chrome-and-ie-browser</a>,chrome会将query进行 encoded再发送请求;而ie原封不动的发送.</li>
925      * <li>由于暂时不能辨别是否encoded过,所以先强制decode再encode;</li>
926      * <li>此处不能先转decode(query,charsetType),参数就是想传 =是转义符</li>
927      * </ol>
928      * </blockquote>
929      * 
930      * @param value
931      *            the value
932      * @param charsetType
933      *            字符编码,建议使用 {@link CharsetType} 定义好的常量<br>
934      *            <span style="color:green">如果是null或者 empty,那么参数部分原样返回,自行处理兼容性问题</span><br>
935      *            否则会先解码,再加码,因为ie浏览器和chrome浏览器 url中访问路径 ,带有中文情况下不一致
936      * @return 如果 <code>value</code>是 null或者empty,返回 {@link StringUtils#EMPTY}<br>
937      *         如果<code>charsetType</code>是 null或者empty,直接返回 <code>value</code><br>
938      *         否则先 {@link URIUtil#decode(String, String)} 再 {@link URIUtil#encode(String, String)}值
939      * @see <a
940      *      href="http://stackoverflow.com/questions/15004593/java-request-getquerystring-value-different-between-chrome-and-ie-browser">
941      *      java-request-getquerystring-value-different-between-chrome-and-ie-browser</a>
942      * @since 1.4.0
943      */
944     private static String decodeAndEncode(String value,String charsetType){
945         if (isNullOrEmpty(value)){
946             return EMPTY;
947         }
948         return isNullOrEmpty(charsetType) ? value : encode(decode(value, charsetType), charsetType);
949     }

951     /**
952      * 拼接url.
953      *
954      * @param beforePathWithoutQueryString
955      *            the before path without query string
956      * @param arrayValueMap
957      *            the array value map
958      * @param charsetType
959      *            字符编码,建议使用 {@link CharsetType} 定义好的常量<br>
960      *            <span style="color:green">如果是null或者 empty,那么参数部分原样返回,自行处理兼容性问题</span><br>
961      *            否则会先解码,再加码,因为ie浏览器和chrome浏览器 url中访问路径 ,带有中文情况下不一致
962      * @return 如果 <code>beforePathWithoutQueryString</code> 是null或者empty,返回 {@link StringUtils#EMPTY}<br>
963      *         如果<code>arrayValueMap</code> 是null或者empty,返回 <code>beforePathWithoutQueryString</code>
964      * @since 1.4.0
965      */
966     static String combineUrl(String beforePathWithoutQueryString,Map<String, String[]> arrayValueMap,String charsetType){
967         if (isNullOrEmpty(beforePathWithoutQueryString)){
968             return EMPTY;
969         }
970         if (isNullOrEmpty(arrayValueMap)){//没有参数 直接return
971             return beforePathWithoutQueryString;
972         }
973 
974         StringBuilder sb = new StringBuilder();
975         sb.append(beforePathWithoutQueryString);
976         sb.append(QUESTIONMARK);
977         sb.append(toQueryStringUseArrayValueMap(toSafeArrayValueMap(arrayValueMap, charsetType)));
978 
979         return sb.toString();
980     }
