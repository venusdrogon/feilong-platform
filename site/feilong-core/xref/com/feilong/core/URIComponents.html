16  package com.feilong.core;
17  
18  /**
19   * uri中使用的字符常量.
20   * 
21   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
22   * @see "org.springframework.web.util.HierarchicalUriComponents.Type"
23   * @see "org.springframework.web.util.UriComponentsBuilder"
24   * @see "org.springframework.web.util.UriComponents"
25   * @since 1.0.9
26   */
27  public final class URIComponents{
28  
29      /** 查询片段 <code>{@value}</code>. */
30      public static final String FRAGMENT     = "#";
31  
32      /** <code>{@value}</code> The question mark is used as a separator and is not part of the query string. */
33      public static final String QUESTIONMARK = "?";
34  
35      /** The Constant ampersand<code>{@value}</code>. */
36      public static final String AMPERSAND    = "&";
37  
38      /**
39       * http协议<code>{@value}</code>.
40       * 
41       * <p>
42       * 超文本传输协议(HTTP,HyperText Transfer Protocol)是互联网上应用最为广泛的一种网络协议。
43       * </p>
44       */
45      public static final String SCHEME_HTTP  = "http";
46  
47      /**
48       * https协议<code>{@value}</code>.
49       * 
50       * <p>
51       * 全称：Hyper Text Transfer Protocol over Secure Socket Layer; <br>
52       * 是以安全为目标的HTTP通道,简单讲是HTTP的安全版。<br>
53       * 即HTTP下加入SSL层,HTTPS的安全基础是SSL,因此加密的详细内容就需要SSL。
54       * </p>
55       * 
56       * <h3>HTTPS和HTTP的区别主要为以下四点:</h3>
57       * 
58       * <blockquote>
59       * <ol>
60       * <li>https协议需要到ca申请证书,一般免费证书很少,需要交费。</li>
61       * <li>http是超文本传输协议,信息是明文传输,https 则是具有安全性的ssl加密传输协议。</li>
62       * <li>http和https使用的是完全不同的连接方式,用的端口也不一样,前者是80,后者是443。</li>
63       * <li>http的连接很简单,是无状态的；HTTPS协议是由SSL+HTTP协议构建的可进行加密传输、身份认证的网络协议,比http协议安全。</li>
64       * </ol>
65       * </blockquote>
66       */
67      public static final String SCHEME_HTTPS = "https";
68  
69      /** Don't let anyone instantiate this class. */
70      private URIComponents(){
71          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
72          //see 《Effective Java》 2nd
73          throw new AssertionError("No " + getClass().getName() + " instances for you!");
74      }
75  }
