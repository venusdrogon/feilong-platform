124     /**
125      * url中的特殊字符转为16进制代码,用于url传递.
126      * 
127      * @param specialCharacter
128      *            特殊字符
129      * @return 特殊字符url编码
130      * @deprecated 将来会重构
131      */
132     @Deprecated
133     public static String specialCharToHexString(String specialCharacter){
134 
135         Map<String, String> specialCharacterMap = new HashMap<String, String>();
136 
137         specialCharacterMap.put("+", "%2B");// URL 中+号表示空格
138         specialCharacterMap.put(" ", "%20");// URL中的空格可以用+号或者编码
139         specialCharacterMap.put("/", "%2F");// 分隔目录和子目录
140         specialCharacterMap.put(QUESTIONMARK, "%3F");// 分隔实际的 URL 和参数
141         specialCharacterMap.put("%", "%25");// 指定特殊字符
142         specialCharacterMap.put("#", "%23");// 表示书签
143         specialCharacterMap.put(AMPERSAND, "%26");// URL 中指定的参数间的分隔符
144         specialCharacterMap.put("=", "%3D");// URL 中指定参数的值
145 
146         if (specialCharacterMap.containsKey(specialCharacter)){
147             return specialCharacterMap.get(specialCharacter);
148         }
149         // 不是 url 特殊字符 原样输出
150         return specialCharacter;
151     }
