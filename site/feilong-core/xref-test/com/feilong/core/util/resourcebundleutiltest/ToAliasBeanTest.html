38  public class ToAliasBeanTest{
39  
40      /**
41       * Test to alias bean.
42       */
43      @Test
44      public void testToAliasBean(){
45          DangaMemCachedConfig dangaMemCachedConfig = toAliasBean(getResourceBundle("messages.memcached"), DangaMemCachedConfig.class);
46          assertThat(
47                          dangaMemCachedConfig,
48                          allOf(//
49                                          hasProperty("serverList", arrayContaining("172.20.3-1.23", "11211", "172.20.31.22", "11211")),
50                                          hasProperty("poolName", is("sidsock2")),
51                                          hasProperty("expireTime", is(180)),
52                                          hasProperty("weight", arrayContaining(2)),
53                                          hasProperty("initConnection", is(10)),
54                                          hasProperty("minConnection", is(5)),
55                                          hasProperty("maxConnection", is(250)),
56                                          hasProperty("maintSleep", is(30)),
57                                          hasProperty("nagle", is(false)),
58                                          hasProperty("socketTo", is(3000)),
59                                          hasProperty("aliveCheck", is(false))
60                          //
61                          ));
62      }
63  
64      /**
65       * Test read properties to alias bean1.
66       */
67      @Test
68      public void testToAliasBean1(){
69          ArrayConverter arrayConverter = new ArrayConverter(String[].class, new StringConverter(), 2);
70          char[] allowedChars = { ':' };
71          arrayConverter.setAllowedChars(allowedChars);
72  
73          ConvertUtils.register(arrayConverter, String[].class);
74  
75          DangaMemCachedConfig dangaMemCachedConfig = toAliasBean(getResourceBundle("messages.memcached"), DangaMemCachedConfig.class);
76          assertThat(
77                          dangaMemCachedConfig,
78                          allOf(//
79                                          hasProperty("serverList", arrayContaining("172.20.3-1.23:11211", "172.20.31.22:11211")),
80                                          hasProperty("poolName", is("sidsock2")),
81                                          hasProperty("expireTime", is(180)),
82                                          hasProperty("weight", arrayContaining(2)),
83                                          hasProperty("initConnection", is(10)),
84                                          hasProperty("minConnection", is(5)),
85                                          hasProperty("maxConnection", is(250)),
86                                          hasProperty("maintSleep", is(30)),
87                                          hasProperty("nagle", is(false)),
88                                          hasProperty("socketTo", is(3000)),
89                                          hasProperty("aliveCheck", is(false))
90                          //
91                          ));
92      }
93  
94      /**
