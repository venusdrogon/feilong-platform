40      /**
41       * Test property comparator.
42       */
43      @Test
44      public void testSortByPropertyNamesValueWithNullPropertyValue(){
45          User u_null_id = new User((Long) null);
46          User id12 = new User(12L);
47          User id2 = new User(2L);
48          User u_null = null;
49          User id1 = new User(1L);
50  
51          List<User> list = toList(u_null_id, id12, id2, u_null, id1, u_null_id);
52          sortListByPropertyNamesValue(list, "id");
53          assertThat(list, contains(u_null_id, u_null_id, id1, id2, id12, u_null));
54      }
55  
56      /**
57       * Test property comparator1.
58       */
59      @Test
60      public void testSortByPropertyNamesValue(){
61          User id12 = new User(12L, 18);
62          User id2 = new User(2L, 36);
63          User id5 = new User(5L, 22);
64          User id1 = new User(1L, 8);
65          List<User> list = toList(id12, id2, id5, id1);
66          sortListByPropertyNamesValue(list, "id");
67  
68          assertThat(list, contains(id1, id2, id5, id12));
69      }

74      @Test
75      public void testSortByPropertyNamesValue2PropertyNames(){
76          User id12_age18 = new User(12L, 18);
77          User id1_age8 = new User(1L, 8);
78          User id2_age30 = new User(2L, 30);
79          User id2_age2 = new User(2L, 2);
80          User id2_age36 = new User(2L, 36);
81          List<User> list = toList(id12_age18, id2_age36, id2_age2, id2_age30, id1_age8);

83          sortListByPropertyNamesValue(list, "id", "age");
84          assertThat(list, contains(id1_age8, id2_age2, id2_age30, id2_age36, id12_age18));
