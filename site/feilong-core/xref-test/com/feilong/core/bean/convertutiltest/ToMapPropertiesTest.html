17  
18  import static com.feilong.core.bean.ConvertUtil.toMap;
19  import static com.feilong.core.util.MapUtil.newLinkedHashMap;
20  import static java.util.Collections.emptyMap;
21  import static org.hamcrest.CoreMatchers.allOf;
22  import static org.hamcrest.Matchers.hasEntry;
23  import static org.junit.Assert.assertEquals;
24  import static org.junit.Assert.assertThat;
25  
26  import java.util.Map;
27  import java.util.Properties;
28  
29  import org.hamcrest.Matchers;
30  import org.junit.Test;
31  
32  /**
33   * The Class ConvertUtilToMapPropertiesTest.
34   *
35   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
36   */
37  public class ToMapPropertiesTest{
38  
39      //*********************com.feilong.core.bean.ConvertUtil.toMap(Properties)*************
40  
41      /**
42       * Test to properties 1.
43       */
44      @Test
45      public void testToProperties1(){
46          Properties properties = new Properties();
47  
48          properties.setProperty("name", "feilong");
49          properties.setProperty("age", "18");
50          properties.setProperty("country", "china");
51  
52          assertThat(
53                          toMap(properties),
54                          allOf(//
55                                          hasEntry("name", "feilong"),
56                                          hasEntry("age", "18"),
57                                          hasEntry("country", "china")));
58      }
59  
60      /**
61       * Test to map empty properties.
62       */
63      @Test
64      public void testToMapEmptyProperties(){
65          assertEquals(emptyMap(), toMap(new Properties()));
66      }
67  
68      /**
69       * Test to map null properties.
70       */
71      @Test
72      public void testToMapNullProperties(){
73          toMap((Properties) null);
74      }
75  
76      /**
77       * Test to properties.
78       */
79      @Test
80      public void testToProperties(){
81          Map<String, Object> map = newLinkedHashMap();
82  
83          map.put("name", "feilong");
84          map.put("age", 18);
85          map.put("country", "china");
86  
87          Properties properties = org.apache.commons.collections4.MapUtils.toProperties(map);
88  
89          assertThat(
90                          toMap(properties),
91                          allOf(//
92                                          Matchers.<String, Object> hasEntry("name", "feilong"),
93                                          Matchers.<String, Object> hasEntry("age", 18),
94                                          Matchers.<String, Object> hasEntry("country", "china")));
95      }
96  }
