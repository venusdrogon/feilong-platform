16  package com.feilong.core.bean.convertutiltest;
17  
18  import static com.feilong.core.bean.ConvertUtil.toIterator;
19  import static com.feilong.core.bean.ConvertUtil.toMap;
20  import static com.feilong.core.util.CollectionsUtil.newArrayList;
21  import static org.hamcrest.Matchers.contains;
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.assertThat;
24  
25  import java.util.Enumeration;
26  import java.util.Iterator;
27  import java.util.List;
28  import java.util.Map;
29  import java.util.StringTokenizer;
30  
31  import org.apache.commons.collections4.IteratorUtils;
32  import org.junit.Test;
33  
34  /**
35   * The Class ConvertUtilToIteratorTest.
36   *
37   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
38   */
39  public class ToIteratorTest{
40  
41      /**
42       * Test to iterator enumeration.
43       */
44      @Test
45      public void testToIteratorEnumeration(){
46          Enumeration<Object> enumeration = new StringTokenizer("this is a test");
47  
48          Iterator<String> iterator = toIterator(enumeration);
49          assertThat(IteratorUtils.toList(iterator), contains("this", "is", "a", "test"));
50      }
51  
52      /**
53       * Test to iterator collection.
54       */
55      @Test
56      public void testToIteratorCollection(){
57          List<String> list = newArrayList();
58          list.add("aaaa");
59          list.add("nnnnn");
60  
61          Iterator<String> iterator = toIterator(list);
62          assertThat(IteratorUtils.toList(iterator), contains("aaaa", "nnnnn"));
63      }
64  
65      /**
66       * Test to iterator string.
67       */
68      @Test
69      public void testToIteratorString(){
70          Iterator<String> iterator = toIterator("1,2");
71          assertThat(IteratorUtils.toList(iterator), contains("1", "2"));
72      }
73  
74      /**
75       * Test to iterator array.
76       */
77      @Test
78      public void testToIteratorArray(){
79          Object[] array = { "5", 8 };
80  
81          Iterator<Object> iterator = toIterator(array);
82          assertThat(IteratorUtils.toList(iterator), org.hamcrest.Matchers.<Object> contains("5", 8));
83      }
84  
85      /**
86       * Test to iterator primitive array.
87       */
88      @Test
89      public void testToIteratorPrimitiveArray(){
90          int[] i2 = { 1, 2 };
91          Iterator<Integer> iterator = toIterator(i2);
92          assertThat(IteratorUtils.toList(iterator), contains(1, 2));
93      }
94  
95      /**
96       * Test to iterator map.
97       */
98      @Test
99      public void testToIteratorMap(){
100         Map<String, String> map = toMap("a", "1", "b", "2");
101 
102         Iterator<String> iterator = toIterator(map);
103         assertThat(IteratorUtils.toList(iterator), contains("1", "2"));
104     }
105 
106     /**
107      * Test to iterator null.
108      */
109     @Test
110     public void testToIteratorNull(){
111         assertEquals(null, toIterator(null));
112     }
113 
114 }
