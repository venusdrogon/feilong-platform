67      /**
68       * 查找子字符串在 字符串中出现的次数.
69       * 
70       * <pre class="code">
71       * StringUtil.searchTimes("xin", "xin")     = 1
72       * StringUtil.searchTimes("xiiiin", "ii")   = 2
73       * </pre>
74       *
75       * @param source
76       *            查找的源字符串
77       * @param target
78       *            目标子串
79       * @return count of target string in source
80       * @see org.apache.commons.lang3.StringUtils#countMatches(CharSequence, CharSequence)
81       * @since 1.0.2
82       * @deprecated 使用 {@link org.apache.commons.lang3.StringUtils#countMatches(CharSequence, CharSequence)}
83       */
84      @Deprecated
85      private static int searchTimes(String source,String target){
86          Validate.notNull(source, "source can't be null!");
87          Validate.notNull(target, "target can't be null!");
88          // times 计数器
89          int count = 0;
90          // while 循环 点
91          int j = 0;
92          // 开始搜索的索引位置
93          int fromIndex = 0;
94          int sourceLength = source.length();
95          // 刚开始从 0的地方查找起
96          while (j != sourceLength){
97              // 从指定的索引开始 返回索引位置
98              int i = source.indexOf(target, fromIndex);
99              if (i != StringUtils.INDEX_NOT_FOUND){
100                 int targetLength = target.length();
101                 // 一旦发现 查找到,下次 循环从找到的地方开始循环
102                 // 查找 从 找到的地方 开始查找
103                 j = i + targetLength;
104                 fromIndex = i + targetLength;
105                 count++;
106             }else{
107                 // 如果发现找不到了 就退出循环
108                 break;
109             }
110         }
111         return count;
112     }

169     /**
170      * [截取]:从第一次出现字符串位置开始(包含)截取到最后,shift表示向前或者向后挪动位数.
171      * 
172      * <h3>示例:</h3>
173      * 
174      * <blockquote>
175      * 
176      * <pre class="code">
177      * StringUtil.substring("jinxin.feilong",".",0)     =   ".feilong"
178      * StringUtil.substring("jinxin.feilong",".",1)     =   "feilong"
179      * </pre>
180      * 
181      * </blockquote>
182      *
183      * @param text
184      *            text
185      * @param beginString
186      *            beginString
187      * @param shift
188      *            负数表示向前,整数表示向后,0表示依旧从自己的位置开始算起
189      * @return 如果 <code>text</code> 是null或者empty,返回 {@link StringUtils#EMPTY}<br>
190      *         如果 <code>beginString</code> 是null或者empty,返回 {@link StringUtils#EMPTY}<br>
191      *         如果 text.indexOf(beginString)==-1,返回 {@link StringUtils#EMPTY}<br>
192      *         如果{@code  beginIndex + shift < 0},抛出 {@link IllegalArgumentException}<br>
193      *         如果{@code  beginIndex + shift > text.length()},返回 {@link StringUtils#EMPTY}<br>
194      *         否则返回 text.substring(beginIndex + shift)<br>
195      * @see org.apache.commons.lang3.StringUtils#substringAfter(String, String)
196      */
197     public static String substring(final String text,String beginString,int shift){
198         if (isNullOrEmpty(text) || isNullOrEmpty(beginString)){
199             return StringUtils.EMPTY;
200         }
201 
202         int beginIndex = text.indexOf(beginString);
203         if (beginIndex == StringUtils.INDEX_NOT_FOUND){// 查不到指定的字符串
204             return StringUtils.EMPTY;
205         }
206         //****************************************************
207         int startIndex = beginIndex + shift;
208         Validate.isTrue(startIndex >= 0, Slf4jUtil.format("[{}] index[{}]+shift[{}]<0,text[{}]", beginString, beginIndex, shift, text));
209 
210         int textLength = text.length();
211         if (startIndex > textLength){
212             LOGGER.warn("beginString [{}] index[{}]+shift[{}]>text[{}].length()[{}]", beginString, beginIndex, shift, text, textLength);
213             return StringUtils.EMPTY;
214         }
215         return text.substring(startIndex);// 索引从0开始
216     }

234     /**
235      * 带有数字格式的数字字符串,与数字相加(一般生成流水号使用).
236      * 
237      * <pre class="code">
238      * StringUtil.stringAddInt("002",2)            =   004
239      * StringUtil.stringAddInt("000002",1200)      =   001202
240      * </pre>
241      * 
242      * @param str
243      *            带有数字格式的数字字符串 比如 002
244      * @param i
245      *            数字
246      * @return 带有数字格式的数字字符串,与数字相加(一般生成流水号使用).
247      * @see NumberUtil#toString(Number, String)
248      */
249     public static String stringAddInt(String str,int i){
250         String pattern = "";
251         for (int j = 0, z = str.length(); j < z; ++j){
252             pattern += "0";
253         }
254         return NumberUtil.toString(Integer.parseInt(str) + i, pattern);
255     }
