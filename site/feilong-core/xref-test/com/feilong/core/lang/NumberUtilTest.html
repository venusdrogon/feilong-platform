43       * int类型转换成16进制字符串.
44       * 
45       * @param i
46       *            int值
47       * @return int类型转换成16进制字符串
48       */

53      /**
54       * 16进制字符串转成int类型.
55       * 
56       * @param hexString
57       *            16进制字符串
58       * @return int类型
59       */
60      public static int hexStringToInt(String hexString){
61          return Integer.parseInt(hexString, 16);
62      }

192     public void toString1(){
193         String pattern = "#######.########";
194         assertEquals("88.02", NumberUtil.toString(88.02, pattern));
195         assertEquals("88.02", NumberUtil.toString(88.020, pattern));
196         assertEquals("88.02002", NumberUtil.toString(88.02002, pattern));
197         assertEquals("88", NumberUtil.toString(88, pattern));
198         assertEquals("88.02000005", NumberUtil.toString(88.02000005, pattern));
199         assertEquals("88.025", NumberUtil.toString(88.02500000, pattern));
200         assertEquals("88.0200005", NumberUtil.toString(88.0200005, pattern));
201         assertEquals("88.002", NumberUtil.toString(88.002, pattern));
202         // //######0
203         LOGGER.debug(NumberUtil.toString((double) 5 / 8 * 100, "#######.###"));
204         DecimalFormat df = new DecimalFormat("######0");
205         LOGGER.debug(df.format(((double) 5 / Integer.valueOf(8)) * 100));
206 
207         assertEquals("C00000008", NumberUtil.toString(8, "C00000000"));
208 
209         assertEquals("24%", NumberUtil.toString(0.24f, NumberPattern.PERCENT_WITH_NOPOINT));
210         assertEquals("24.00%", NumberUtil.toString(0.24f, NumberPattern.PERCENT_WITH_2POINT));
211 
212         Integer a = 1;
213         Long b = 400L;
214         assertEquals("0.25%", NumberUtil.toString((double) a / b, NumberPattern.PERCENT_WITH_2POINT));
215 
216         assertEquals("1", NumberUtil.toString(0.8, NumberPattern.NO_SCALE));
217         assertEquals("-1", NumberUtil.toString(-0.8, NumberPattern.NO_SCALE));
218         assertEquals("-2", NumberUtil.toString(-1.8, NumberPattern.NO_SCALE));
219         assertEquals("2", NumberUtil.toString(1.8, NumberPattern.NO_SCALE));
220         assertEquals("111112", NumberUtil.toString(111111.5, NumberPattern.NO_SCALE));
221         assertEquals("111113", NumberUtil.toString(111112.5, NumberPattern.NO_SCALE));
222         assertEquals("88888888", NumberUtil.toString(88888888, NumberPattern.NO_SCALE));
223     }
