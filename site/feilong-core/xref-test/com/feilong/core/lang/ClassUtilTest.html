156     /**
157      * 获得 class info map for LOGGER.
158      *
159      * @param klass
160      *            the klass
161      * @return 如果 <code>klass</code> 是null或者empty,返回 {@link Collections#emptyMap()}<br>
162      */
163     public static Map<String, Object> getClassInfoMapForLog(Class<?> klass){
164         if (isNullOrEmpty(klass)){
165             return Collections.emptyMap();
166         }
167 
168         Map<String, Object> map = new LinkedHashMap<String, Object>();
169 
170         map.put("clz.getCanonicalName()", klass.getCanonicalName());//"com.feilong.core.date.DatePattern"
171         map.put("clz.getName()", klass.getName());//"com.feilong.core.date.DatePattern"
172         map.put("clz.getSimpleName()", klass.getSimpleName());//"DatePattern"
173 
174         map.put("clz.getComponentType()", klass.getComponentType());
175         // 类是不是"基本类型". 基本类型,包括void和boolean、byte、char、short、int、long、float 和 double这几种类型.
176         map.put("clz.isPrimitive()", klass.isPrimitive());
177 
178         // 类是不是"本地类".本地类,就是定义在方法内部的类.
179         map.put("clz.isLocalClass()", klass.isLocalClass());
180         // 类是不是"成员类".成员类,是内部类的一种,但是它不是"内部类"或"匿名类".
181         map.put("clz.isMemberClass()", klass.isMemberClass());
182 
183         //isSynthetic()是用来判断Class是不是"复合类".这在java应用程序中只会返回false,不会返回true.因为,JVM中才会产生复合类,在java应用程序中不存在"复合类"！
184         map.put("clz.isSynthetic()", klass.isSynthetic());
185         map.put("clz.isArray()", klass.isArray());
186         map.put("clz.isAnnotation()", klass.isAnnotation());
187 
188         //当且仅当这个类是匿名类此方法返回true.
189         map.put("clz.isAnonymousClass()", klass.isAnonymousClass());
190         map.put("clz.isEnum()", klass.isEnum());
191 
192         return map;
193     }
